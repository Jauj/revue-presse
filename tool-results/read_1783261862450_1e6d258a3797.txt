    30→    30→export async function saveDailyMemory(env, date, reviewContent, articles) {
    31→    31→  const dateStr = date.toISOString().split('T')[0];
    32→    32→  const key = `${MEM_PREFIX}day:${dateStr}`;
    33→    33→
    34→    34→  const themes = extractThemesFromReview(reviewContent);
    35→    35→  const sources = [...new Set(articles.map(a => a.sourceName))];
    36→    36→  const keyFacts = extractKeyFacts(reviewContent);
    37→    37→  const actors = extractActors(reviewContent);
    38→    38→
    39→    39→  const memory = {
    40→    40→    date: dateStr,
    41→    41→    themes,
    42→    42→    sources,
    43→    43→    keyFacts,
    44→    44→    actors,
    45→    45→    articleCount: articles.length,
    46→    46→    sourceCount: sources.length,
    47→    47→    wordEstimate: reviewContent.split(/\s+/).length,
    48→    48→    createdAt: new Date().toISOString(),
    49→    49→  };
    50→    50→
    51→    51→  await env.CACHE.put(key, JSON.stringify(memory), { expirationTtl: 30 * 24 * 3600 });
    52→    52→  return memory;
    53→    53→}

    60→    60→export async function updateSemanticMemory(env, dailyMemory) {
    61→    61→  const key = `${MEM_PREFIX}themes`;
    62→    62→  const raw = await env.CACHE.get(key);
    63→    63→  let semantic = raw ? JSON.parse(raw) : { themes: {}, lastUpdated: null, totalDays: 0 };
    64→    64→
    65→    65→  for (const theme of dailyMemory.themes) {
    66→    66→    const normalized = normalizeTheme(theme);
    67→    67→    if (!semantic.themes[normalized]) {
    68→    68→      semantic.themes[normalized] = {
    69→    69→        name: theme,
    70→    70→        count: 0,
    71→    71→        firstSeen: dailyMemory.date,
    72→    72→        lastSeen: dailyMemory.date,
    73→    73→        relatedSources: [],
    74→    74→        relatedFacts: [],
    75→    75→        days: [],
    76→    76→      };
    77→    77→    }
    78→    78→    semantic.themes[normalized].count++;
    79→    79→    semantic.themes[normalized].lastSeen = dailyMemory.date;
    80→    80→    if (!semantic.themes[normalized].days.includes(dailyMemory.date)) {
    81→    81→      semantic.themes[normalized].days.push(dailyMemory.date);
    82→    82→    }
    83→    83→
    84→    84→    const existing = new Set(semantic.themes[normalized].relatedSources);
    85→    85→    for (const s of dailyMemory.sources) existing.add(s);
    86→    86→    semantic.themes[normalized].relatedSources = [...existing].slice(0, 10);
    87→    87→
    88→    88→    for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
    89→    89→      if (!semantic.themes[normalized].relatedFacts.includes(fact)) {
    90→    90→        semantic.themes[normalized].relatedFacts.push(fact);
    91→    91→      }
    92→    92→    }
    93→    93→    semantic.themes[normalized].relatedFacts =
    94→    94→      semantic.themes[normalized].relatedFacts.slice(-5);
    95→    95→  }
    96→    96→
    97→    97→  semantic.totalDays++;
    98→    98→  semantic.lastUpdated = new Date().toISOString();
    99→    99→
   100→   100→  await env.CACHE.put(key, JSON.stringify(semantic), { expirationTtl: 365 * 24 * 3600 });
   101→   101→  return semantic;
   102→   102→}

   105→   105→// COUCHE 3 — NARRATIVES (fil rouge inter-jours)
   106→   106→// KV: memory:narratives  (TTL 90j)
   107→   107→// Détecte les récits qui traversent plusieurs jours

   109→   109→
   110→   110→export async function updateNarratives(env, dailyMemory) {
   111→   111→  const key = `${MEM_PREFIX}narratives`;
   112→   112→  const raw = await env.CACHE.get(key);
   113→   113→  let narratives = raw ? JSON.parse(raw) : { active: [], archived: [], lastUpdated: null };
   114→   114→
   115→   115→  const todayThemes = dailyMemory.themes.map(t => t.toLowerCase());
   116→   116→  let touchedNarratives = [];
   117→   117→
   118→   118→  // 1. Essayer de rattacher les thèmes du jour à des narratives existantes
   119→   119→  for (const narrative of narratives.active) {
   120→   120→    let matched = false;
   121→   121→    for (const theme of todayThemes) {
   122→   122→      const similarity = computeThemeSimilarity(theme, narrative.title.toLowerCase());
   123→   123→      if (similarity > 0.4) {
   124→   124→        // Mettre à jour la narrative
   125→   125→        narrative.lastSeen = dailyMemory.date;
   126→   126→        narrative.dayCount = (narrative.dayCount || 1) + 1;
   127→   127→        narrative.days = narrative.days || [];
   128→   128→        if (!narrative.days.includes(dailyMemory.date)) {
   129→   129→          narrative.days.push(dailyMemory.date);
   130→   130→        }
   131→   131→        // Ajouter les nouveaux faits
   132→   132→        for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
   133→   133→          if (!narrative.facts.some(f => f.fact === fact)) {
   134→   134→            narrative.facts.push({ fact, date: dailyMemory.date });
   135→   135→          }
   136→   136→        }
   137→   137→        // Ajouter les sources
   138→   138→        const existingSources = new Set(narrative.sources || []);
   139→   139→        for (const s of dailyMemory.sources) existingSources.add(s);
   140→   140→        narrative.sources = [...existingSources];
   141→   141→        // Dernière évolution
   142→   142→        narrative.lastEvolution = dailyMemory.themes[0] || '';
   143→   143→        matched = true;
   144→   144→        touchedNarratives.push(narrative.id);
   145→   145→        break;
   146→   146→      }
   147→   147→    }
   148→   148→  }
   149→   149→
   150→   150→  // 2. Créer de nouvelles narratives pour les thèmes non rattachés
   151→   151→  for (const theme of dailyMemory.themes.slice(0, 3)) {
   152→   152→    const themeLower = theme.toLowerCase();
   153→   153→    const alreadyTracked = touchedNarratives.some(id => {
   154→   154→      const n = narratives.active.find(a => a.id === id);
   155→   155→      return n && computeThemeSimilarity(themeLower, n.title.toLowerCase()) > 0.5;
   156→   156→    });
   157→   157→    if (!alreadyTracked) {
   158→   158→      narratives.active.push({
   159→   159→        id: `nar_${dailyMemory.date}_${normalizeTheme(theme).substring(0, 20)}`,
   160→   160→        title: theme,
   161→   161→        firstSeen: dailyMemory.date,
   162→   162→        lastSeen: dailyMemory.date,
   163→   163→        dayCount: 1,
   164→   164→        days: [dailyMemory.date],
   165→   165→        facts: dailyMemory.keyFacts.slice(0, 2).map(f => ({ fact: f, date: dailyMemory.date })),
   166→   166→        sources: dailyMemory.sources.slice(0, 5),
   167→   167→        lastEvolution: theme,
   168→   168→        status: 'active',
   169→   169→      });
   170→   170→    }
   171→   171→  }
   172→   172→
   173→   173→  // 3. Archiver les narratives inactives (>7 jours sans update)
   174→   174→  narratives.active = narratives.active.filter(n => {
   175→   175→    const daysSince = daysBetween(n.lastSeen, dailyMemory.date);
   176→   176→    if (daysSince > 7 && n.dayCount < 3) return false; // supprimer les éphémères
   177→   177→    if (daysSince > 14) {
   178→   178→      n.status = 'archived';
   179→   179→      n.archivedAt = dailyMemory.date;
   180→   180→      narratives.archived = narratives.archived || [];
   181→   181→      narratives.archived.push(n);
   182→   182→      return false;
   183→   183→    }
   184→   184→    return true;
   185→   185→  });
   186→   186→
   187→   187→  // Garder max 10 narratives actives (les plus récentes)
   188→   188→  narratives.active.sort((a, b) => b.lastSeen.localeCompare(a.lastSeen));
   189→   189→  narratives.active = narratives.active.slice(0, 10);
   190→   190→
   191→   191→  // Garder max 20 narratives archivées
   192→   192→  if (narratives.archived) {
   193→   193→    narratives.archived = narratives.archived.slice(-20);
   194→   194→  }
   195→   195→
   196→   196→  narratives.lastUpdated = new Date().toISOString();
   197→   197→
   198→   198→  await env.CACHE.put(key, JSON.stringify(narratives), { expirationTtl: 90 * 24 * 3600 });
   199→   199→  return narratives;
   200→   200→}

   204→   204→// KV: memory:distilled:2026-07  (TTL 365j)
   205→   205→// "Le rêve" : l'IA analyse N jours de mémoires et produit

   209→   209→/**
   210→   210→ * Processus de rêve complet (distillation IA)
   211→   211→ * Lit les mémoires des derniers N jours et produit des insights
   212→   212→ */
   213→   213→export async function dreamDistill(env, daysBack = 14, force = false) {
   214→   214→  const today = new Date();
   215→   215→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   216→   216→  const distilledKey = `${MEM_PREFIX}distilled:${monthKey}`;
   217→   217→
   218→   218→  // Vérifier si on a déjà distillé récemment (pas plus d'une fois par 3 jours)
   219→   219→  if (!force) {
   220→   220→    const lastDreamRaw = await env.CACHE.get(`${MEM_PREFIX}last_dream`);
   221→   221→    if (lastDreamRaw) {
   222→   222→      const lastDream = JSON.parse(lastDreamRaw);
   223→   223→      const daysSinceDream = daysBetween(lastDream.date, today.toISOString().split('T')[0]);
   224→   224→      if (daysSinceDream < 3) {
   225→   225→        return { skipped: true, reason: `Dernier rêve il y a ${daysSinceDream}j (min 3)`, lastDream };
   226→   226→      }
   227→   227→    }
   228→   228→  }
   229→   229→
   230→   230→  // 1. Collecter les mémoires épisodiques
   231→   231→  const dailyMemories = [];
   232→   232→  for (let i = 1; i <= daysBack; i++) {
   233→   233→    const d = new Date(today);
   234→   234→    d.setDate(d.getDate() - i);
   235→   235→    const dateStr = d.toISOString().split('T')[0];
   236→   236→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   237→   237→    if (raw) dailyMemories.push(JSON.parse(raw));
   238→   238→  }
   239→   239→
   240→   240→  if (dailyMemories.length < 3) {
   241→   241→    return { skipped: true, reason: `Seulement ${dailyMemories.length} mémoires (min 3)`, hasEnough: false };
   242→   242→  }
   243→   243→
   244→   244→  // 2. Collecter les narratives
   245→   245→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   246→   246→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   247→   247→
   248→   248→  // 3. Construire le matériel de rêve (compact)
   249→   249→  const dreamMaterial = buildDreamMaterial(dailyMemories, narratives);
   250→   250→
   251→   251→  // 4. Appel IA de distillation
   252→   252→  const startTime = Date.now();
   253→   253→  let distilled = null;
   254→   254→  try {
   255→   255→    distilled = await callAIDream(env, dreamMaterial);
   256→   256→  } catch (err) {
   257→   257→    console.error(`Rêve IA échoué: ${err.message}`);
   258→   258→    // Fallback : distillation rule-based
   259→   259→    distilled = ruleBasedDistillation(dailyMemories, narratives);
   260→   260→  }
   261→   261→  const dreamDuration = Date.now() - startTime;
   262→   262→
   263→   263→  // 5. Stocker la distillation mensuelle
   264→   264→  const distilledData = {
   265→   265→    month: monthKey,
   266→   266→    date: today.toISOString().split('T')[0],
   267→   267→    daysAnalyzed: dailyMemories.length,
   268→   268→    dreamDurationMs: dreamDuration,
   269→   269→    aiPowered: !!distilled.aiGenerated,
   270→   270→    ...distilled,
   271→   271→  };
   272→   272→
   273→   273→  await env.CACHE.put(distilledKey, JSON.stringify(distilledData), { expirationTtl: 365 * 24 * 3600 });
   274→   274→
   275→   275→  // 6. Mettre à jour la mémoire éditoriale
   276→   276→  await updateEditorialMemory(env, dailyMemories, distilled);
   277→   277→
   278→   278→  // 7. Enregistrer le dernier rêve
   279→   279→  await env.CACHE.put(`${MEM_PREFIX}last_dream`, JSON.stringify({
   280→   280→    date: today.toISOString().split('T')[0],
   281→   281→    type: dailyMemories.length >= 7 ? 'deep' : 'light',
   282→   282→    duration: dreamDuration,
   283→   283→    daysAnalyzed: dailyMemories.length,
   284→   284→    monthKey,
   285→   285→  }), { expirationTtl: 30 * 24 * 3600 });
   286→   286→
   287→   287→  // 8. Nettoyer les vieilles mémoires épisodiques (au-delà de 14 jours)
   288→   288→  await pruneOldMemories(env, 14);
   289→   289→
   290→   290→  return {
   291→   291→    success: true,
   292→   292→    daysAnalyzed: dailyMemories.length,
   293→   293→    dreamDurationMs: dreamDuration,
   294→   294→    monthKey,
   295→   295→    distilledThemes: distilled.patterns?.length || 0,
   296→   296→    narrativeCount: narratives?.active?.length || 0,
   297→   297→  };
   298→   298→}

   301→   301→// COUCHE 5 — MÉMOIRE ÉDITORIALE (préférences apprises)
   302→   302→// KV: memory:editorial  (TTL 365j)
   303→   303→// Apprend quels patterns fonctionnent bien
   304→   304→// ============================================================
   305→   305→
   306→   306→async function updateEditorialMemory(env, dailyMemories, distilled) {
   307→   307→  const key = `${MEM_PREFIX}editorial`;
   308→   308→  const raw = await env.CACHE.get(key);
   309→   309→  let editorial = raw ? JSON.parse(raw) : {
   310→   310→    sourcePatterns: {},
   311→   311→    topicBalance: {},
   312→   312→    avgArticleCount: 0,
   313→   313→    avgThemesPerDay: 0,
   314→   314→    totalReviews: 0,
   315→   315→    preferences: [],
   316→   316→    lastUpdated: null,
   317→   317→  };
   318→   318→
   319→   319→  // Statistiques glissantes
   320→   320→  editorial.totalReviews++;
   321→   321→  const recentCounts = dailyMemories.slice(-7).map(m => m.articleCount);
   322→   322→  editorial.avgArticleCount = Math.round(recentCounts.reduce((a, b) => a + b, 0) / recentCounts.length);
   323→   323→  const recentThemes = dailyMemories.slice(-7).map(m => m.themes.length);
   324→   324→  editorial.avgThemesPerDay = Math.round(recentThemes.reduce((a, b) => a + b, 0) / recentThemes.length);
   325→   325→
   326→   326→  // Patterns par source
   327→   327→  for (const mem of dailyMemories) {
   328→   328→    for (const source of mem.sources) {
   329→   329→      if (!editorial.sourcePatterns[source]) {
   330→   330→        editorial.sourcePatterns[source] = { appearances: 0, themesContributed: [] };
   331→   331→      }
   332→   332→      editorial.sourcePatterns[source].appearances++;
   333→   333→      for (const theme of mem.themes.slice(0, 2)) {
   334→   334→        const existing = new Set(editorial.sourcePatterns[source].themesContributed);
   335→   335→        existing.add(theme);
   336→   336→        editorial.sourcePatterns[source].themesContributed = [...existing].slice(0, 10);
   337→   337→      }
   338→   338→    }
   339→   339→  }
   340→   340→
   341→   341→  // Préférences apprises (depuis la distillation IA)
   342→   342→  if (distilled.insights) {
   343→   343→    for (const insight of distilled.insights.slice(0, 5)) {
   344→   344→      if (!editorial.preferences.some(p => p.insight === insight)) {
   345→   345→        editorial.preferences.push({
   346→   346→          insight,
   347→   347→          learnedFrom: dailyMemories[dailyMemories.length - 1]?.date,
   348→   348→          createdAt: new Date().toISOString(),
   349→   349→        });
   350→   350→      }
   351→   351→    }
   352→   352→    editorial.preferences = editorial.preferences.slice(-10);
   353→   353→  }
   354→   354→
   355→   355→  editorial.lastUpdated = new Date().toISOString();
   356→   356→  await env.CACHE.put(key, JSON.stringify(editorial), { expirationTtl: 365 * 24 * 3600 });
   357→   357→  return editorial;
   358→   358→}

   363→   363→
   364→   364→export async function getMemoryContext(env, maxDays = 7) {
   365→   365→  const today = new Date();
   366→   366→  const dailyMemories = [];
   367→   367→  const themesRaw = await env.CACHE.get(`${MEM_PREFIX}themes`);
   368→   368→  const semantic = themesRaw ? JSON.parse(themesRaw) : null;
   369→   369→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   370→   370→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   371→   371→  const editorialRaw = await env.CACHE.get(`${MEM_PREFIX}editorial`);
   372→   372→  const editorial = editorialRaw ? JSON.parse(editorialRaw) : null;
   373→   373→
   374→   374→  // Dernière distillation
   375→   375→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   376→   376→  const lastMonthKey = today.getMonth() === 0
   377→   377→    ? `${today.getFullYear() - 1}-12`
   378→   378→    : `${today.getFullYear()}-${String(today.getMonth()).padStart(2, '0')}`;
   379→   379→  const distilledRaw = await env.CACHE.get(`${MEM_PREFIX}distilled:${monthKey}`)
   380→   380→    || await env.CACHE.get(`${MEM_PREFIX}distilled:${lastMonthKey}`);
   381→   381→  const distilled = distilledRaw ? JSON.parse(distilledRaw) : null;
   382→   382→
   383→   383→  // Lire les derniers jours de mémoire épisodique
   384→   384→  for (let i = 1; i <= maxDays; i++) {
   385→   385→    const d = new Date(today);
   386→   386→    d.setDate(d.getDate() - i);
   387→   387→    const dateStr = d.toISOString().split('T')[0];
   388→   388→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   389→   389→    if (raw) dailyMemories.push(JSON.parse(raw));
   390→   390→  }
   391→   391→
   392→   392→  if (dailyMemories.length === 0 && !semantic && !distilled) {
   393→   393→    return null;
   394→   394→  }
   395→   395→
   396→   396→  return buildRichContextString(dailyMemories, semantic, narratives, distilled, editorial);
   397→   397→}

   400→   400→// NETTOYAGE — Supprimer les anciennes mémoires
   401→   401→// ============================================================
   402→   402→
   403→   403→export async function pruneOldMemories(env, keepDays = 14) {
   404→   404→  const today = new Date();
   405→   405→  let pruned = 0;
   406→   406→
   407→   407→  for (let i = keepDays + 1; i <= 60; i++) {
   408→   408→    const d = new Date(today);
   409→   409→    d.setDate(d.getDate() - i);
   410→   410→    const dateStr = d.toISOString().split('T')[0];
   411→   411→    const key = `${MEM_PREFIX}day:${dateStr}`;
   412→   412→    const existing = await env.CACHE.get(key);
   413→   413→    if (existing) {
   414→   414→      await env.CACHE.delete(key);
   415→   415→      pruned++;
   416→   416→    }
   417→   417→  }
   418→   418→
   419→   419→  return pruned;
   420→   420→}

   428→   428→  const stats = {
   429→   429→    totalDailyMemories: 0,
   430→   430→    semanticThemes: 0,
   431→   431→    activeNarratives: 0,
   432→   432→    hasDistillation: false,

   654→   654→function ruleBasedDistillation(dailyMemories, narratives) {
   655→   655→  // Patterns récurrents
   656→   656→  const themeFreq = {};
   657→   657→  for (const mem of dailyMemories) {
   658→   658→    for (const t of mem.themes) {
   659→   659→      const norm = normalizeTheme(t);
   660→   660→      themeFreq[norm] = (themeFreq[norm] || 0) + 1;
   661→   661→    }
   662→   662→  }
   663→   663→
   664→   664→  const patterns = Object.entries(themeFreq)
   665→   665→    .sort((a, b) => b[1] - a[1])
   666→   666→    .slice(0, 6)
   667→   667→    .map(([key, count]) => ({
   668→   668→      type: 'recurring',
   669→   669→      theme: key.replace(/_/g, ' '),
   670→   670→      frequency: `${count}/${dailyMemories.length} jours`,
   671→   671→      trend: count > dailyMemories.length * 0.5 ? 'stable' : 'emerging',
   672→   672→      note: `Présent ${count} jours sur ${dailyMemories.length}`,
   673→   673→    }));
   674→   674→
   675→   675→  // Narratives depuis la mémoire narratives
   676→   676→  const narList = (narratives?.active || []).slice(0, 4).map(n => ({
   677→   677→    title: n.title,
   678→   678→    arc: `${n.dayCount} jours d'évolution`,
   679→   679→    key_moments: n.facts?.slice(-2).map(f => f.fact) || [],
   680→   680→    status: n.dayCount >= 5 ? 'peaked' : 'active',
   681→   681→  }));
   682→   682→
   683→   683→  // Source dynamics
   684→   684→  const sourceFreq = {};
   685→   685→  for (const mem of dailyMemories) {
   686→   686→    for (const s of mem.sources) {
   687→   687→      sourceFreq[s] = (sourceFreq[s] || 0) + 1;
   688→   688→    }
   689→   689→  }
   690→   690→  const sourceDynamics = Object.entries(sourceFreq)
   691→   691→    .sort((a, b) => b[1] - a[1])
   692→   692→    .slice(0, 5)
   693→   693→    .map(([source, count]) => ({
   694→   694→      source,
   695→   695→      role: count > dailyMemories.length * 0.7 ? 'leader' : 'contributor',
   696→   696→      notes: `Présent ${count} jours`,
   697→   697→    }));
   698→   698→
   699→   699→  return {
   700→   700→    patterns,
   701→   701→    narratives: narList,
   702→   702→    insights: [],
   703→   703→    source_dynamics: sourceDynamics,
   704→   704→    blind_spots: [],
   705→   705→    macro_trends: [],
   706→   706→    aiGenerated: false,
   707→   707→    generatedAt: new Date().toISOString(),
   708→   708→  };
   709→   709→}

   811→   811→function extractThemesFromReview(reviewText) {
   812→   812→  const themes = [];
   813→   813→  const sectionRegex = /\*\*([^*]{3,40})\*\*\s*[:：]/g;
   814→   814→  let match;
   815→   815→  while ((match = sectionRegex.exec(reviewText)) !== null) {
   816→   816→    const theme = match[1].trim();
   817→   817→    const nonThemes = [
   818→   818→      "L'ESSENTIEL DU JOUR", "ANALYSE THÉMATIQUE", "TENDANCES",
   819→   819→      "CHIFFRES CLÉS", "À SURVEILLER", "ESSENTIEL", "TENDANCES & PERSPECTIVES",
   820→   820→    ];
   821→   821→    if (!nonThemes.some(nt => theme.toUpperCase().includes(nt.toUpperCase()))) {
   822→   822→      themes.push(theme);
   823→   823→    }
   824→   824→  }
   825→   825→
   826→   826→  const axiomRegex = /\*\*([^*]{2,20})\s*[:：]\*\*/g;
   827→   827→  while ((match = axiomRegex.exec(reviewText)) !== null) {
   828→   828→    const axiom = match[1].trim();
   829→   829→    if (axiom.length > 3 && axiom.length < 25 && !themes.includes(axiom)) {
   830→   830→      themes.push(axiom);
   831→   831→    }
   832→   832→  }
   833→   833→
   834→   834→  return [...new Set(themes)].slice(0, 10);
   835→   835→}

   837→   837→function extractKeyFacts(reviewText) {
   838→   838→  const facts = [];

   840→   840→  let match;
   841→   841→  while ((match = factRegex.exec(reviewText)) !== null) {
   842→   842→    const fact = match[1].trim();
   843→   843→    if (/\d+/.test(fact)) facts.push(fact);
   844→   844→  }
   845→   845→  return facts.slice(0, 10);

   848→   848→function extractActors(reviewText) {
   849→   849→  const actors = new Set();
   850→   850→  // Patterns communs : noms propres après "Selon", "D'après", en gras
   851→   851→  const selonRegex = /(?:Selon|D'après)\s+\*\*([^*]{2,40})\*\*/g;
   852→   852→  let match;
   853→   853→  while ((match = selonRegex.exec(reviewText)) !== null) {

   872→   872→function computeThemeSimilarity(a, b) {
   873→   873→  const wordsA = new Set(a.split(/[\s_]+/));
   874→   874→  const wordsB = new Set(b.split(/[\s_]+/));
   875→   875→  if (wordsA.size === 0 || wordsB.size === 0) return 0;
   876→   876→  let intersection = 0;
   877→   877→  for (const w of wordsA) {
   878→   878→    if (wordsB.has(w)) intersection++;
   879→   879→  }
   880→   880→  return intersection / Math.max(wordsA.size, wordsB.size);
   881→   881→}
