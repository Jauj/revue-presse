    30→    30→export async function saveDailyMemory(env, date, reviewContent, articles) {
    31→    31→  const dateStr = date.toISOString().split('T')[0];
    32→    32→  const key = `${MEM_PREFIX}day:${dateStr}`;
    33→    33→
    34→    34→  const themes = extractThemesFromReview(reviewContent);
    35→    35→  const sources = [...new Set(articles.map(a => a.sourceName))];
    36→    36→  const keyFacts = extractKeyFacts(reviewContent);
    37→    37→  const actors = extractActors(reviewContent);
    38→    38→  const entities = extractEntities(reviewContent);
    39→    39→
    40→    40→  const memory = {
    41→    41→    date: dateStr,
    42→    42→    themes,
    43→    43→    sources,
    44→    44→    keyFacts,
    45→    45→    actors,
    46→    46→    entities,
    47→    47→    articleCount: articles.length,
    48→    48→    sourceCount: sources.length,
    49→    49→    wordEstimate: reviewContent.split(/\s+/).length,
    50→    50→    createdAt: new Date().toISOString(),
    51→    51→  };
    52→    52→
    53→    53→  await env.CACHE.put(key, JSON.stringify(memory), { expirationTtl: 30 * 24 * 3600 });
    54→    54→
    55→    55→  // Mettre à jour l'index des jours
    56→    56→  await updateDayIndex(env, dateStr);
    57→    57→
    58→    58→  return memory;
    59→    59→}

    66→    66→export async function updateSemanticMemory(env, dailyMemory) {
    67→    67→  const key = `${MEM_PREFIX}themes`;
    68→    68→  const raw = await env.CACHE.get(key);
    69→    69→  let semantic = raw ? JSON.parse(raw) : { themes: {}, lastUpdated: null, totalDays: 0 };
    70→    70→
    71→    71→  for (const theme of dailyMemory.themes) {
    72→    72→    const normalized = normalizeTheme(theme);
    73→    73→    if (!semantic.themes[normalized]) {
    74→    74→      semantic.themes[normalized] = {
    75→    75→        name: theme,
    76→    76→        count: 0,
    77→    77→        firstSeen: dailyMemory.date,
    78→    78→        lastSeen: dailyMemory.date,
    79→    79→        relatedSources: [],
    80→    80→        relatedFacts: [],
    81→    81→        days: [],
    82→    82→      };
    83→    83→    }
    84→    84→    semantic.themes[normalized].count++;
    85→    85→    semantic.themes[normalized].lastSeen = dailyMemory.date;
    86→    86→    if (!semantic.themes[normalized].days.includes(dailyMemory.date)) {
    87→    87→      semantic.themes[normalized].days.push(dailyMemory.date);
    88→    88→    }
    89→    89→
    90→    90→    const existing = new Set(semantic.themes[normalized].relatedSources);
    91→    91→    for (const s of dailyMemory.sources) existing.add(s);
    92→    92→    semantic.themes[normalized].relatedSources = [...existing].slice(0, 10);
    93→    93→
    94→    94→    for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
    95→    95→      if (!semantic.themes[normalized].relatedFacts.includes(fact)) {
    96→    96→        semantic.themes[normalized].relatedFacts.push(fact);
    97→    97→      }
    98→    98→    }
    99→    99→    semantic.themes[normalized].relatedFacts =
   100→   100→      semantic.themes[normalized].relatedFacts.slice(-5);
   101→   101→  }
   102→   102→
   103→   103→  semantic.totalDays++;
   104→   104→  semantic.lastUpdated = new Date().toISOString();
   105→   105→
   106→   106→  await env.CACHE.put(key, JSON.stringify(semantic), { expirationTtl: 365 * 24 * 3600 });
   107→   107→  return semantic;
   108→   108→}

   116→   116→export async function updateNarratives(env, dailyMemory) {
   117→   117→  const key = `${MEM_PREFIX}narratives`;
   118→   118→  const raw = await env.CACHE.get(key);
   119→   119→  let narratives = raw ? JSON.parse(raw) : { active: [], archived: [], lastUpdated: null };
   120→   120→
   121→   121→  const todayThemes = dailyMemory.themes.map(t => t.toLowerCase());
   122→   122→  let touchedNarratives = [];
   123→   123→
   124→   124→  // 1. Essayer de rattacher les thèmes du jour à des narratives existantes
   125→   125→  for (const narrative of narratives.active) {
   126→   126→    let matched = false;
   127→   127→    for (const theme of todayThemes) {
   128→   128→      const similarity = computeThemeSimilarity(theme, narrative.title.toLowerCase());
   129→   129→      if (similarity > 0.4) {
   130→   130→        // Mettre à jour la narrative
   131→   131→        narrative.lastSeen = dailyMemory.date;
   132→   132→        narrative.dayCount = (narrative.dayCount || 1) + 1;
   133→   133→        narrative.days = narrative.days || [];
   134→   134→        if (!narrative.days.includes(dailyMemory.date)) {
   135→   135→          narrative.days.push(dailyMemory.date);
   136→   136→        }
   137→   137→        // Ajouter les nouveaux faits
   138→   138→        for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
   139→   139→          if (!narrative.facts.some(f => f.fact === fact)) {
   140→   140→            narrative.facts.push({ fact, date: dailyMemory.date });
   141→   141→          }
   142→   142→        }
   143→   143→        // Ajouter les sources
   144→   144→        const existingSources = new Set(narrative.sources || []);
   145→   145→        for (const s of dailyMemory.sources) existingSources.add(s);
   146→   146→        narrative.sources = [...existingSources];
   147→   147→        // Dernière évolution
   148→   148→        narrative.lastEvolution = dailyMemory.themes[0] || '';
   149→   149→        matched = true;
   150→   150→        touchedNarratives.push(narrative.id);
   151→   151→        break;
   152→   152→      }
   153→   153→    }
   154→   154→  }
   155→   155→
   156→   156→  // 2. Créer de nouvelles narratives pour les thèmes non rattachés
   157→   157→  for (const theme of dailyMemory.themes.slice(0, 3)) {
   158→   158→    const themeLower = theme.toLowerCase();
   159→   159→    const alreadyTracked = touchedNarratives.some(id => {
   160→   160→      const n = narratives.active.find(a => a.id === id);
   161→   161→      return n && computeThemeSimilarity(themeLower, n.title.toLowerCase()) > 0.5;
   162→   162→    });
   163→   163→    if (!alreadyTracked) {
   164→   164→      narratives.active.push({
   165→   165→        id: `nar_${dailyMemory.date}_${normalizeTheme(theme).substring(0, 20)}`,
   166→   166→        title: theme,
   167→   167→        firstSeen: dailyMemory.date,
   168→   168→        lastSeen: dailyMemory.date,
   169→   169→        dayCount: 1,
   170→   170→        days: [dailyMemory.date],
   171→   171→        facts: dailyMemory.keyFacts.slice(0, 2).map(f => ({ fact: f, date: dailyMemory.date })),
   172→   172→        sources: dailyMemory.sources.slice(0, 5),
   173→   173→        lastEvolution: theme,
   174→   174→        status: 'active',
   175→   175→      });
   176→   176→    }
   177→   177→  }
   178→   178→
   179→   179→  // 3. Archiver les narratives inactives (>7 jours sans update)
   180→   180→  narratives.active = narratives.active.filter(n => {
   181→   181→    const daysSince = daysBetween(n.lastSeen, dailyMemory.date);
   182→   182→    if (daysSince > 7 && n.dayCount < 3) return false; // supprimer les éphémères
   183→   183→    if (daysSince > 14) {
   184→   184→      n.status = 'archived';
   185→   185→      n.archivedAt = dailyMemory.date;
   186→   186→      narratives.archived = narratives.archived || [];
   187→   187→      narratives.archived.push(n);
   188→   188→      return false;
   189→   189→    }
   190→   190→    return true;
   191→   191→  });
   192→   192→
   193→   193→  // Garder max 10 narratives actives (les plus récentes)
   194→   194→  narratives.active.sort((a, b) => b.lastSeen.localeCompare(a.lastSeen));
   195→   195→  narratives.active = narratives.active.slice(0, 10);
   196→   196→
   197→   197→  // Garder max 20 narratives archivées
   198→   198→  if (narratives.archived) {
   199→   199→    narratives.archived = narratives.archived.slice(-20);
   200→   200→  }
   201→   201→
   202→   202→  narratives.lastUpdated = new Date().toISOString();
   203→   203→
   204→   204→  await env.CACHE.put(key, JSON.stringify(narratives), { expirationTtl: 90 * 24 * 3600 });
   205→   205→  return narratives;
   206→   206→}

   210→   210→// KV: memory:distilled:2026-07  (TTL 365j)

   215→   215→/**
   216→   216→ * Processus de rêve complet (distillation IA)
   217→   217→ * Lit les mémoires des derniers N jours et produit des insights
   218→   218→ */
   219→   219→export async function dreamDistill(env, daysBack = 14, force = false) {
   220→   220→  const today = new Date();
   221→   221→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   222→   222→  const distilledKey = `${MEM_PREFIX}distilled:${monthKey}`;
   223→   223→
   224→   224→  // Vérifier si on a déjà distillé récemment (pas plus d'une fois par 3 jours)
   225→   225→  if (!force) {
   226→   226→    const lastDreamRaw = await env.CACHE.get(`${MEM_PREFIX}last_dream`);
   227→   227→    if (lastDreamRaw) {
   228→   228→      const lastDream = JSON.parse(lastDreamRaw);
   229→   229→      const daysSinceDream = daysBetween(lastDream.date, today.toISOString().split('T')[0]);
   230→   230→      if (daysSinceDream < 3) {
   231→   231→        return { skipped: true, reason: `Dernier rêve il y a ${daysSinceDream}j (min 3)`, lastDream };
   232→   232→      }
   233→   233→    }
   234→   234→  }
   235→   235→
   236→   236→  // 1. Collecter les mémoires épisodiques via day_index (optimisation KV)
   237→   237→  const dailyMemories = [];
   238→   238→  const indexRaw = await env.CACHE.get(`${MEM_PREFIX}day_index`);
   239→   239→  if (indexRaw) {
   240→   240→    const index = JSON.parse(indexRaw);
   241→   241→    const cutoff = new Date(today);
   242→   242→    cutoff.setDate(cutoff.getDate() - daysBack);
   243→   243→    const cutoffStr = cutoff.toISOString().split('T')[0];
   244→   244→    const recentDays = (index.days || []).filter(d => d <= today.toISOString().split('T')[0] && d >= cutoffStr);
   245→   245→    for (const dateStr of recentDays) {
   246→   246→      const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   247→   247→      if (raw) dailyMemories.push(JSON.parse(raw));
   248→   248→    }
   249→   249→  } else {
   250→   250→    // Fallback : lecture séquentielle si pas d'index
   251→   251→    for (let i = 1; i <= daysBack; i++) {
   252→   252→      const d = new Date(today);
   253→   253→      d.setDate(d.getDate() - i);
   254→   254→      const dateStr = d.toISOString().split('T')[0];
   255→   255→      const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   256→   256→      if (raw) dailyMemories.push(JSON.parse(raw));
   257→   257→    }
   258→   258→  }
   259→   259→
   260→   260→  if (dailyMemories.length < 3) {
   261→   261→    return { skipped: true, reason: `Seulement ${dailyMemories.length} mémoires (min 3)`, hasEnough: false };
   262→   262→  }
   263→   263→
   264→   264→  // 2. Collecter les narratives
   265→   265→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   266→   266→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   267→   267→
   268→   268→  // 3. Construire le matériel de rêve (compact)
   269→   269→  const dreamMaterial = buildDreamMaterial(dailyMemories, narratives);
   270→   270→
   271→   271→  // 4. Appel IA de distillation
   272→   272→  const startTime = Date.now();
   273→   273→  let distilled = null;
   274→   274→  try {
   275→   275→    distilled = await callAIDream(env, dreamMaterial);
   276→   276→  } catch (err) {
   277→   277→    console.error(`Rêve IA échoué: ${err.message}`);
   278→   278→    // Fallback : distillation rule-based
   279→   279→    distilled = ruleBasedDistillation(dailyMemories, narratives);
   280→   280→  }
   281→   281→  const dreamDuration = Date.now() - startTime;
   282→   282→
   283→   283→  // 5. Stocker la distillation mensuelle
   284→   284→  const distilledData = {
   285→   285→    month: monthKey,
   286→   286→    date: today.toISOString().split('T')[0],
   287→   287→    daysAnalyzed: dailyMemories.length,
   288→   288→    dreamDurationMs: dreamDuration,
   289→   289→    aiPowered: !!distilled.aiGenerated,
   290→   290→    ...distilled,
   291→   291→  };
   292→   292→
   293→   293→  await env.CACHE.put(distilledKey, JSON.stringify(distilledData), { expirationTtl: 365 * 24 * 3600 });
   294→   294→
   295→   295→  // 6. Mettre à jour la mémoire éditoriale
   296→   296→  await updateEditorialMemory(env, dailyMemories, distilled);
   297→   297→
   298→   298→  // 7. Enregistrer le dernier rêve
   299→   299→  await env.CACHE.put(`${MEM_PREFIX}last_dream`, JSON.stringify({
   300→   300→    date: today.toISOString().split('T')[0],
   301→   301→    type: dailyMemories.length >= 7 ? 'deep' : 'light',
   302→   302→    duration: dreamDuration,
   303→   303→    daysAnalyzed: dailyMemories.length,
   304→   304→    monthKey,
   305→   305→  }), { expirationTtl: 30 * 24 * 3600 });
   306→   306→
   307→   307→  // 8. Nettoyer les vieilles mémoires épisodiques (au-delà de 14 jours)
   308→   308→  await pruneOldMemories(env, 14);
   309→   309→
   310→   310→  return {
   311→   311→    success: true,
   312→   312→    daysAnalyzed: dailyMemories.length,
   313→   313→    dreamDurationMs: dreamDuration,
   314→   314→    monthKey,
   315→   315→    distilledThemes: distilled.patterns?.length || 0,
   316→   316→    narrativeCount: narratives?.active?.length || 0,
   317→   317→  };
   318→   318→}

   321→   321→// COUCHE 5 — MÉMOIRE ÉDITORIALE (préférences apprises)
   322→   322→// KV: memory:editorial  (TTL 365j)
   323→   323→// Apprend quels patterns fonctionnent bien

   325→   325→
   326→   326→async function updateEditorialMemory(env, dailyMemories, distilled) {
   327→   327→  const key = `${MEM_PREFIX}editorial`;
   328→   328→  const raw = await env.CACHE.get(key);
   329→   329→  let editorial = raw ? JSON.parse(raw) : {
   330→   330→    sourcePatterns: {},
   331→   331→    topicBalance: {},
   332→   332→    avgArticleCount: 0,
   333→   333→    avgThemesPerDay: 0,
   334→   334→    totalReviews: 0,
   335→   335→    preferences: [],
   336→   336→    lastUpdated: null,
   337→   337→  };
   338→   338→
   339→   339→  // Statistiques glissantes
   340→   340→  editorial.totalReviews++;
   341→   341→  const recentCounts = dailyMemories.slice(-7).map(m => m.articleCount);
   342→   342→  editorial.avgArticleCount = Math.round(recentCounts.reduce((a, b) => a + b, 0) / recentCounts.length);
   343→   343→  const recentThemes = dailyMemories.slice(-7).map(m => m.themes.length);
   344→   344→  editorial.avgThemesPerDay = Math.round(recentThemes.reduce((a, b) => a + b, 0) / recentThemes.length);
   345→   345→
   346→   346→  // Patterns par source
   347→   347→  for (const mem of dailyMemories) {
   348→   348→    for (const source of mem.sources) {
   349→   349→      if (!editorial.sourcePatterns[source]) {
   350→   350→        editorial.sourcePatterns[source] = { appearances: 0, themesContributed: [] };
   351→   351→      }
   352→   352→      editorial.sourcePatterns[source].appearances++;
   353→   353→      for (const theme of mem.themes.slice(0, 2)) {
   354→   354→        const existing = new Set(editorial.sourcePatterns[source].themesContributed);
   355→   355→        existing.add(theme);
   356→   356→        editorial.sourcePatterns[source].themesContributed = [...existing].slice(0, 10);
   357→   357→      }
   358→   358→    }
   359→   359→  }
   360→   360→
   361→   361→  // Préférences apprises (depuis la distillation IA)
   362→   362→  if (distilled.insights) {
   363→   363→    for (const insight of distilled.insights.slice(0, 5)) {
   364→   364→      if (!editorial.preferences.some(p => p.insight === insight)) {
   365→   365→        editorial.preferences.push({
   366→   366→          insight,
   367→   367→          learnedFrom: dailyMemories[dailyMemories.length - 1]?.date,
   368→   368→          createdAt: new Date().toISOString(),
   369→   369→        });
   370→   370→      }
   371→   371→    }
   372→   372→    editorial.preferences = editorial.preferences.slice(-10);
   373→   373→  }
   374→   374→
   375→   375→  editorial.lastUpdated = new Date().toISOString();
   376→   376→  await env.CACHE.put(key, JSON.stringify(editorial), { expirationTtl: 365 * 24 * 3600 });
   377→   377→  return editorial;
   378→   378→}

   383→   383→
   384→   384→export async function getMemoryContext(env, maxDays = 7) {
   385→   385→  const today = new Date();
   386→   386→  const dailyMemories = [];
   387→   387→  const themesRaw = await env.CACHE.get(`${MEM_PREFIX}themes`);
   388→   388→  const semantic = themesRaw ? JSON.parse(themesRaw) : null;
   389→   389→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   390→   390→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   391→   391→  const editorialRaw = await env.CACHE.get(`${MEM_PREFIX}editorial`);
   392→   392→  const editorial = editorialRaw ? JSON.parse(editorialRaw) : null;
   393→   393→
   394→   394→  // Dernière distillation
   395→   395→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   396→   396→  const lastMonthKey = today.getMonth() === 0
   397→   397→    ? `${today.getFullYear() - 1}-12`
   398→   398→    : `${today.getFullYear()}-${String(today.getMonth()).padStart(2, '0')}`;
   399→   399→  const distilledRaw = await env.CACHE.get(`${MEM_PREFIX}distilled:${monthKey}`)
   400→   400→    || await env.CACHE.get(`${MEM_PREFIX}distilled:${lastMonthKey}`);
   401→   401→  const distilled = distilledRaw ? JSON.parse(distilledRaw) : null;
   402→   402→
   403→   403→  // Lire les derniers jours de mémoire épisodique via day_index
   404→   404→  const indexRaw = await env.CACHE.get(`${MEM_PREFIX}day_index`);
   405→   405→  if (indexRaw) {
   406→   406→    const index = JSON.parse(indexRaw);
   407→   407→    const cutoff = new Date(today);
   408→   408→    cutoff.setDate(cutoff.getDate() - maxDays);
   409→   409→    const cutoffStr = cutoff.toISOString().split('T')[0];
   410→   410→    const todayStr = today.toISOString().split('T')[0];
   411→   411→    const recentDays = (index.days || []).filter(d => d <= todayStr && d >= cutoffStr);
   412→   412→    for (const dateStr of recentDays) {
   413→   413→      const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   414→   414→      if (raw) dailyMemories.push(JSON.parse(raw));
   415→   415→    }
   416→   416→  } else {
   417→   417→    for (let i = 1; i <= maxDays; i++) {
   418→   418→      const d = new Date(today);
   419→   419→      d.setDate(d.getDate() - i);
   420→   420→      const dateStr = d.toISOString().split('T')[0];
   421→   421→      const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   422→   422→      if (raw) dailyMemories.push(JSON.parse(raw));
   423→   423→    }
   424→   424→  }
   425→   425→
   426→   426→  if (dailyMemories.length === 0 && !semantic && !distilled) {
   427→   427→    return null;
   428→   428→  }
   429→   429→

   542→   542→function buildDreamMaterial(dailyMemories, narratives) {
   543→   543→  let material = `## MATÉRIEL DE RÊVE — ${dailyMemories.length} jours de mémoires\n\n`;
   544→   544→
   545→   545→  // Résumé chronologique compact
   546→   546→  material += '### Chronologie (du plus ancien au plus récent)\n';
   547→   547→  for (const mem of dailyMemories) {
   548→   548→    const themesStr = mem.themes.slice(0, 3).join(' | ');
   549→   549→    material += `- ${mem.date}: ${themesStr} (${mem.articleCount} articles, ${mem.sourceCount} sources)\n`;
   550→   550→  }
   551→   551→
   552→   552→  // Thèmes avec fréquence
   553→   553→  const themeFreq = {};
   554→   554→  for (const mem of dailyMemories) {
   555→   555→    for (const t of mem.themes) {
   556→   556→      const norm = normalizeTheme(t);
   557→   557→      themeFreq[norm] = (themeFreq[norm] || 0) + 1;
   558→   558→    }
   559→   559→  }
   560→   560→  const frequent = Object.entries(themeFreq)
   561→   561→    .sort((a, b) => b[1] - a[1])
   562→   562→    .slice(0, 12);
   563→   563→
   564→   564→  material += '\n### Thèmes par fréquence\n';
   565→   565→  for (const [key, count] of frequent) {
   566→   566→    const bar = '█'.repeat(Math.min(count, 10));
   567→   567→    material += `- ${key.replace(/_/g, ' ')} : ${bar} (${count}/${dailyMemories.length} jours)\n`;
   568→   568→  }
   569→   569→
   570→   570→  // Sources les plus présentes
   571→   571→  const sourceFreq = {};
   572→   572→  for (const mem of dailyMemories) {
   573→   573→    for (const s of mem.sources) {
   574→   574→      sourceFreq[s] = (sourceFreq[s] || 0) + 1;
   575→   575→    }
   576→   576→  }
   577→   577→  const topSources = Object.entries(sourceFreq)
   578→   578→    .sort((a, b) => b[1] - a[1])
   579→   579→    .slice(0, 8);
   580→   580→
   581→   581→  material += '\n### Sources les plus citées\n';
   582→   582→  for (const [source, count] of topSources) {
   583→   583→    material += `- ${source} : présente ${count}/${dailyMemories.length} jours\n`;
   584→   584→  }
   585→   585→
   586→   586→  // Faits clés par jour
   587→   587→  material += '\n### Faits marquants par jour\n';
   588→   588→  for (const mem of dailyMemories.slice(-5)) {
   589→   589→    material += `\n**${mem.date}**\n`;
   590→   590→    for (const fact of mem.keyFacts.slice(0, 3)) {
   591→   591→      material += `  - ${fact}\n`;
   592→   592→    }
   593→   593→  }
   594→   594→
   595→   595→  // Narratives actives
   596→   596→  if (narratives?.active?.length > 0) {
   597→   597→    material += '\n### Narratives en cours\n';
   598→   598→    for (const nar of narratives.active.slice(0, 5)) {
   599→   599→      material += `- **${nar.title}** : ${nar.dayCount} jours, dernier le ${nar.lastSeen}\n`;
   600→   600→      if (nar.facts?.length > 0) {
   601→   601→        material += `  Évolution: ${nar.facts[nar.facts.length - 1].fact}\n`;
   602→   602→      }
   603→   603→    }
   604→   604→  }
   605→   605→
   606→   606→  // Acteurs récurrents
   607→   607→  const actorFreq = {};
   608→   608→  for (const mem of dailyMemories) {
   609→   609→    if (mem.actors) {
   610→   610→      for (const a of mem.actors) {
   611→   611→        actorFreq[a] = (actorFreq[a] || 0) + 1;
   612→   612→      }
   613→   613→    }
   614→   614→  }
   615→   615→  const topActors = Object.entries(actorFreq)
   616→   616→    .sort((a, b) => b[1] - a[1])
   617→   617→    .slice(0, 8);
   618→   618→
   619→   619→  if (topActors.length > 0) {
   620→   620→    material += '\n### Acteurs récurrents\n';
   621→   621→    for (const [actor, count] of topActors) {
   622→   622→      material += `- ${actor} : ${count} mentions\n`;
   623→   623→    }
   624→   624→  }
   625→   625→
   626→   626→  return material;
   627→   627→}

   698→   698→function ruleBasedDistillation(dailyMemories, narratives) {
   699→   699→  // Patterns récurrents
   700→   700→  const themeFreq = {};
   701→   701→  for (const mem of dailyMemories) {
   702→   702→    for (const t of mem.themes) {
   703→   703→      const norm = normalizeTheme(t);
   704→   704→      themeFreq[norm] = (themeFreq[norm] || 0) + 1;
   705→   705→    }
   706→   706→  }
   707→   707→
   708→   708→  const patterns = Object.entries(themeFreq)
   709→   709→    .sort((a, b) => b[1] - a[1])
   710→   710→    .slice(0, 6)
   711→   711→    .map(([key, count]) => ({
   712→   712→      type: 'recurring',
   713→   713→      theme: key.replace(/_/g, ' '),
   714→   714→      frequency: `${count}/${dailyMemories.length} jours`,
   715→   715→      trend: count > dailyMemories.length * 0.5 ? 'stable' : 'emerging',
   716→   716→      note: `Présent ${count} jours sur ${dailyMemories.length}`,
   717→   717→    }));
   718→   718→
   719→   719→  // Narratives depuis la mémoire narratives
   720→   720→  const narList = (narratives?.active || []).slice(0, 4).map(n => ({
   721→   721→    title: n.title,
   722→   722→    arc: `${n.dayCount} jours d'évolution`,
   723→   723→    key_moments: n.facts?.slice(-2).map(f => f.fact) || [],
   724→   724→    status: n.dayCount >= 5 ? 'peaked' : 'active',
   725→   725→  }));
   726→   726→
   727→   727→  // Source dynamics
   728→   728→  const sourceFreq = {};
   729→   729→  for (const mem of dailyMemories) {
   730→   730→    for (const s of mem.sources) {
   731→   731→      sourceFreq[s] = (sourceFreq[s] || 0) + 1;
   732→   732→    }
   733→   733→  }
   734→   734→  const sourceDynamics = Object.entries(sourceFreq)
   735→   735→    .sort((a, b) => b[1] - a[1])
   736→   736→    .slice(0, 5)
   737→   737→    .map(([source, count]) => ({
   738→   738→      source,
   739→   739→      role: count > dailyMemories.length * 0.7 ? 'leader' : 'contributor',
   740→   740→      notes: `Présent ${count} jours`,
   741→   741→    }));
   742→   742→
   743→   743→  return {
   744→   744→    patterns,
   745→   745→    narratives: narList,
   746→   746→    insights: [],
   747→   747→    source_dynamics: sourceDynamics,
   748→   748→    blind_spots: [],
   749→   749→    macro_trends: [],
   750→   750→    aiGenerated: false,
   751→   751→    generatedAt: new Date().toISOString(),
   752→   752→  };
   753→   753→}

   759→   759→function buildRichContextString(dailyMemories, semantic, narratives, distilled, editorial) {
   760→   760→  let context = '## CONTEXTE MÉMOIRE (mémoire éditoriale long terme)\n\n';
   761→   761→
   762→   762→  // 1. Narratives actives (priorité haute — fil rouge)
   763→   763→  if (narratives?.active?.length > 0) {
   764→   764→    context += '### Récits en cours (narratives)\n';
   765→   765→    context += 'Ces histoires traversent plusieurs jours. Situe l\'actualité du jour par rapport à ces récits :\n\n';
   766→   766→    for (const nar of narratives.active.slice(0, 4)) {
   767→   767→      const statusIcon = nar.dayCount >= 5 ? '🔴' : nar.dayCount >= 3 ? '🟡' : '🟢';
   768→   768→      context += `${statusIcon} **${nar.title}** (${nar.dayCount} jours, dernier: ${nar.lastSeen})\n`;
   769→   769→      if (nar.facts?.length > 0) {
   770→   770→        const lastFact = nar.facts[nar.facts.length - 1];
   771→   771→        context += `  Dernière évolution: ${lastFact.fact}\n`;
   772→   772→      }
   773→   773→    }
   774→   774→    context += '\n';
   775→   775→  }
   776→   776→
   777→   777→  // 2. Connaissances distillées (insights de haut niveau)
   778→   778→  if (distilled) {
   779→   779→    if (distilled.insights?.length > 0) {
   780→   780→      context += '### Connaissances consolidées (rêve)\n';
   781→   781→      context += 'Patterns identifiés par analyse croisée des jours précédents :\n\n';
   782→   782→      for (const insight of distilled.insights.slice(0, 4)) {
   783→   783→        context += `- 💡 ${insight}\n`;
   784→   784→      }
   785→   785→      context += '\n';
   786→   786→    }
   787→   787→
   788→   788→    if (distilled.macro_trends?.length > 0) {
   789→   789→      context += '### Tendances macro\n';
   790→   790→      for (const trend of distilled.macro_trends.slice(0, 3)) {
   791→   791→        context += `- 📈 **${trend.trend}** : ${trend.description}\n`;
   792→   792→      }
   793→   793→      context += '\n';
   794→   794→    }
   795→   795→
   796→   796→    if (distilled.blind_spots?.length > 0) {
   797→   797→      context += '### Angles morts détectés\n';
   798→   798→      for (const bs of distilled.blind_spots.slice(0, 2)) {
   799→   799→        context += `- ⚠️ ${bs}\n`;
   800→   800→      }
   801→   801→      context += '\n';
   802→   802→    }
   803→   803→  }
   804→   804→
   805→   805→  // 3. Thèmes sémantiques récurrents
   806→   806→  if (semantic && Object.keys(semantic.themes).length > 0) {
   807→   807→    context += '### Thèmes récurrents\n';
   808→   808→    const sorted = Object.entries(semantic.themes)
   809→   809→      .sort((a, b) => b[1].count - a[1].count)
   810→   810→      .slice(0, 6);
   811→   811→
   812→   812→    for (const [key, data] of sorted) {
   813→   813→      const trend = data.count > 5 ? 'récurrent' : data.count > 2 ? 'émergent' : 'nouveau';
   814→   814→      const trendIcon = data.count > 5 ? '🔴' : data.count > 2 ? '🟡' : '🟢';
   815→   815→      context += `${trendIcon} **${data.name}** (${data.count}j, ${trend})\n`;
   816→   816→    }
   817→   817→    context += '\n';
   818→   818→  }
   819→   819→
   820→   820→  // 4. Mémoire épisodique récente (derniers 3 jours, compact)
   821→   821→  if (dailyMemories.length > 0) {
   822→   822→    context += '### Revues récentes (derniers jours)\n';
   823→   823→    for (const mem of dailyMemories.slice(0, 3)) {
   824→   824→      const themesStr = mem.themes.slice(0, 3).join(', ') || 'sans thème';
   825→   825→      context += `- **${mem.date}** : ${themesStr} (${mem.articleCount} articles)\n`;
   826→   826→    }
   827→   827→    context += '\n';
   828→   828→  }
   829→   829→
   830→   830→  // 5. Préférences éditoriales apprises
   831→   831→  if (editorial?.preferences?.length > 0) {
   832→   832→    context += '### Préférences apprises\n';
   833→   833→    for (const pref of editorial.preferences.slice(0, 3)) {
   834→   834→      context += `- 📝 ${pref.insight}\n`;
   835→   835→    }
   836→   836→    context += '\n';
   837→   837→  }
   838→   838→
   839→   839→  // Instructions pour l'IA
   840→   840→  context += '### Instructions d\'utilisation de la mémoire\n';
   841→   841→  context += '- Si un sujet fait partie d\'une NARRATIVE en cours, situe l\'évolution dans le récit\n';
   842→   842→  context += '- Mentionne les TENDANCES MACRO quand l\'actualité du jour s\'inscrit dans ce cadre\n';
   843→   843→  context += '- Pour les thèmes RÉCURRENTS, indique si c\'est une continuité ou un changement\n';
   844→   844→  context += '- Si l\'actualité touche un ANGLE MORT identifié, accorde-lui une attention particulière\n';
   845→   845→  context += '- Ne répète PAS les informations déjà détaillées les jours précédents sauf évolution significative\n';
   846→   846→  context += '- Les CONNAISSANCES CONSOLIDÉES sont des patterns fiables — utilise-les pour enrichir l\'analyse\n';
   847→   847→
   848→   848→  return context;

   855→   855→function extractThemesFromReview(reviewText) {
   856→   856→  const themes = [];
   857→   857→  const nonThemes = new Set([
   858→   858→    "L'ESSENTIEL DU JOUR", "ANALYSE THÉMATIQUE", "TENDANCES",
   859→   859→    "CHIFFRES CLÉS", "À SURVEILLER", "ESSENTIEL", "TENDANCES & PERSPECTIVES",
   860→   860→    "SOMMAIRE", "POINTS DE TENSION", "SOURCES", "POUR ALLER PLUS LOIN",
   861→   861→    "APPROFONDIR", "NOTE FINALE",
   862→   862→  ]);
   863→   863→
   864→   864→  // Format éditorial : **1. Titre éditorial** ou **1. Titre : sous-titre**
   865→   865→  const editorialRegex = /^\*\*(\d+)\.[\s]+(.+?)(?:\s*[:：]\s*(.+?))?\*\*$/gm;
   866→   866→  let match;
   867→   867→  while ((match = editorialRegex.exec(reviewText)) !== null) {
   868→   868→    const title = match[2].trim();
   869→   869→    if (title.length > 5 && !nonThemes.has(title.toUpperCase())) {
   870→   870→      themes.push(title);
   871→   871→    }
   872→   872→  }
   873→   873→
   874→   874→  // Format Smart Brevity legacy : **THÈME 1** suivi de :
   875→   875→  const sectionRegex = /\*\*([^*\d]{3,40})\*\*\s*[:：]/g;
   876→   876→  while ((match = sectionRegex.exec(reviewText)) !== null) {
   877→   877→    const theme = match[1].trim();
   878→   878→    if (!nonThemes.has(theme.toUpperCase()) && !themes.includes(theme)) {
   879→   879→      themes.push(theme);
   880→   880→    }
   881→   881→  }
   882→   882→
   883→   883→  // Format axiom legacy : **Bold Axiom :**
   884→   884→  const axiomRegex = /\*\*([^*]{2,25})\s*[:：]\*\*/g;
   885→   885→  while ((match = axiomRegex.exec(reviewText)) !== null) {
   886→   886→    const axiom = match[1].trim();
   887→   887→    if (axiom.length > 3 && axiom.length < 25 && !nonThemes.has(axiom.toUpperCase()) && !themes.includes(axiom)) {
   888→   888→      themes.push(axiom);
   889→   889→    }
   890→   890→  }
   891→   891→
   892→   892→  return [...new Set(themes)].slice(0, 12);
   893→   893→}

   895→   895→function extractKeyFacts(reviewText) {
   896→   896→  const facts = [];
   897→   897→  const seen = new Set();
   898→   898→
   899→   899→  // 1. Lignes à puces substantielles (30-250 chars)
   900→   900→  const bulletRegex = /^-\s+(.{30,250})$/gm;
   901→   901→  let match;
   902→   902→  while ((match = bulletRegex.exec(reviewText)) !== null) {
   903→   903→    const fact = match[1].trim();
   904→   904→    // Exclure les lignes qui sont juste des titres de sections ou des liens
   905→   905→    if (!fact.startsWith('**') && !fact.startsWith('http') && !seen.has(fact.substring(0, 50))) {
   906→   906→      facts.push(fact);
   907→   907→      seen.add(fact.substring(0, 50));
   908→   908→    }
   909→   909→  }
   910→   910→
   911→   911→  // 2. Phrases avec des chiffres précis dans les paragraphes (signal de faits durs)
   912→   912→  const numberRegex = /[^.!?]*(?:\d{2,}|\d+\s*(?:millions?|milliards?|%|morts?|blessés?|personnes|jours|ans))[^.!?]*[.!?]/gi;
   913→   913→  while ((match = numberRegex.exec(reviewText)) !== null) {
   914→   914→    const fact = match[0].trim();
   915→   915→    if (fact.length > 30 && fact.length < 250 && !seen.has(fact.substring(0, 50))) {
   916→   916→      facts.push(fact);
   917→   917→      seen.add(fact.substring(0, 50));
   918→   918→    }
   919→   919→  }
   920→   920→
   921→   921→  return [...new Set(facts)].slice(0, 15);
   922→   922→}

   924→   924→function extractActors(reviewText) {
   925→   925→  const actors = new Set();
   926→   926→  let match;
   927→   927→
   928→   928→  // 1. Noms propres en gras (2+ mots commençant par majuscule)
   929→   929→  const boldNameRegex = /\*\*([A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ][a-zàâéèêëîïôùûüçæœ]+(?:\s+[A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ][a-zàâéèêëîïôùûüçæœ]+){0,3})\*\*/g;
   930→   930→  while ((match = boldNameRegex.exec(reviewText)) !== null) {
   931→   931→    const name = match[1].trim();
   932→   932→    // Exclure les titres de sections (commençant par un chiffre ou trop longs)
   933→   933→    if (!/^\d/.test(name) && name.length < 40 && name.split(/\s+/).length >= 2) {
   934→   934→      actors.add(name);
   935→   935→    }
   936→   936→  }
   937→   937→
   938→   938→  // 2. Après « Selon », « D'après »
   939→   939→  const selonRegex = /(?:Selon|D'après)\s+([^,.]{2,40})/gi;
   940→   940→  while ((match = selonRegex.exec(reviewText)) !== null) {
   941→   941→    const name = match[1].trim();
   942→   942→    // Garder si ça ressemble à un nom propre (majuscule au début)
   943→   943→    if (/^[A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ]/.test(name) && name.length > 2) {
   944→   944→      actors.add(name);
   945→   945→    }
   946→   946→  }
   947→   947→
   948→   948→  // 3. Noms propres entre guillemets suivis d'une source

   950→   950→  while ((match = quoteActorRegex.exec(reviewText)) !== null) {
   951→   951→    actors.add(match[1].trim());
   952→   952→  }
   953→   953→
   954→   954→  // 4. Liste élargie de figures connues (pattern flexible)
   955→   955→  const knownFigures = [
   956→   956→    'Macron', 'Barnier', 'Le Pen', 'Mélenchon', 'Scholz', 'Trump', 'Biden',

   959→   959→    'Von der Leyen', 'Blinken', 'Lavrov', 'Khamenei', 'Modi', 'Meloni',
   960→   960→  ];
   961→   961→  for (const figure of knownFigures) {
   962→   962→    if (reviewText.includes(figure)) actors.add(figure);
   963→   963→  }
   964→   964→
   965→   965→  return [...actors].slice(0, 15);
   966→   966→}

  1057→  1057→function computeThemeSimilarity(a, b) {
  1058→  1058→  const wordsA = new Set(a.split(/[\s_]+/));
  1059→  1059→  const wordsB = new Set(b.split(/[\s_]+/));
  1060→  1060→  if (wordsA.size === 0 || wordsB.size === 0) return 0;
  1061→  1061→  let intersection = 0;
  1062→  1062→  for (const w of wordsA) {
  1063→  1063→    if (wordsB.has(w)) intersection++;
  1064→  1064→  }
  1065→  1065→  return intersection / Math.max(wordsA.size, wordsB.size);
  1066→  1066→}
