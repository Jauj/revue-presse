   241→/** Échappe les caractères XML dangereux pour éviter de casser le parsing */
   242→function escapeXML(str) {
   243→  return String(str)
   244→    .replace(/&/g, '&amp;')
   245→    .replace(/</g, '&lt;')
   246→    .replace(/>/g, '&gt;')
   247→    .replace(/"/g, '&quot;')
   248→    .replace(/'/g, '&apos;');
   249→}

   251→function formatDateForAI(dateStr) {
   252→  if (!dateStr) return 'date inconnue';
   253→  try {
   254→    const d = new Date(dateStr);
   255→    if (isNaN(d.getTime())) return dateStr;
   256→    return d.toLocaleDateString('fr-FR', { day: 'numeric', month: 'long', year: 'numeric', timeZone: 'Europe/Paris' });
   257→  } catch { return dateStr; }
   258→}

   260→function buildArticlesXML(articles, maxWordsPerArticle = 200) {
   261→  let xml = `<articles count="${articles.length}">\n`;
   262→  for (let i = 0; i < articles.length; i++) {
   263→    const a = articles[i];
   264→    const text = a.extractedText || a.description || '(indisponible)';
   265→    const excerpt = text.split(/\s+/).slice(0, maxWordsPerArticle).join(' ');
   266→    xml += `<article id="${i + 1}">\n`;
   267→    xml += `<source>${escapeXML(a.sourceName)}</source>\n`;
   268→    xml += `<lang>${escapeXML(a.sourceLang || 'fr')}</lang>\n`;
   269→    xml += `<category>${escapeXML(a.sourceCategory || 'autre')}</category>\n`;
   270→    xml += `<title>${escapeXML(a.title)}</title>\n`;
   271→    xml += `<pub_date>${escapeXML(formatDateForAI(a.pubDate))}</pub_date>\n`;
   272→    xml += `<content>\n${escapeXML(excerpt)}\n</content>\n`;
   273→    if (a.link) xml += `<url>${escapeXML(a.link)}</url>\n`;
   274→    xml += `</article>\n\n`;
   275→  }
   276→  xml += `</articles>`;
   277→  return xml;
   278→}

   305→/** Appel IA format OpenAI-compatible (Groq, Mistral, etc.) */
   306→async function callOpenAICompatible(endpoint, model, apiKey, systemPrompt, userPrompt, maxTokens = 8000) {
   307→  const response = await fetch(endpoint, {
   308→    method: 'POST',
   309→    headers: { 'Authorization': `Bearer ${apiKey}`, 'Content-Type': 'application/json' },
   310→    body: JSON.stringify({
   311→      model,
   312→      messages: [
   313→        { role: 'system', content: systemPrompt },
   314→        { role: 'user', content: userPrompt },
   315→      ],
   316→      temperature: 0.2,
   317→      max_tokens: maxTokens,
   318→      top_p: 0.9,
   319→    }),
   320→    signal: AbortSignal.timeout(180000),
   321→  });
   322→
   323→  if (!response.ok) {
   324→    const errBody = await response.text().catch(() => '');
   325→    throw new Error(`HTTP ${response.status}: ${errBody.substring(0, 200)}`);
   326→  }
   327→  return (await response.json()).choices[0].message.content;
   328→}

   330→/** Appel Gemini (API native) — timeout 180s */
   331→async function callGemini(apiKey, systemPrompt, userPrompt, maxOutputTokens = 8000) {
   332→  const url = `${GEMINI_ENDPOINT}?key=${apiKey}`;
   333→  const response = await fetch(url, {
   334→    method: 'POST',
   335→    headers: { 'Content-Type': 'application/json' },
   336→    body: JSON.stringify({
   337→      contents: [{
   338→        role: 'user',
   339→        parts: [{ text: `${systemPrompt}\n\n---\n\n${userPrompt}` }],
   340→      }],
   341→      generationConfig: { temperature: 0.2, maxOutputTokens, topP: 0.9 },
   342→    }),
   343→    signal: AbortSignal.timeout(180000),
   344→  });
   345→  if (!response.ok) throw new Error(`Gemini HTTP ${response.status}: ${await response.text()}`);
   346→  const data = await response.json();
   347→  return data.candidates?.[0]?.content?.parts?.[0]?.text || '';
   348→}

   350→/** Workers AI fallback */
   351→async function callWorkersAI(env, prompt) {
   352→  if (!env.AI) throw new Error('Workers AI non configuré');
   353→  const models = ['@cf/meta/llama-4-scout-17b-16e-instruct', '@cf/meta/llama-3.1-8b-instruct'];
   354→  for (const model of models) {
   355→    try {
   356→      const r = await env.AI.run(model, { prompt, max_tokens: 8000, temperature: 0.2 });
   357→      const text = r.response || '';
   358→      if (text.length > 100) return text;
   359→    } catch (e) { /* skip */ }
   360→  }
   361→  throw new Error('Workers AI: tous modèles échoués');
   362→}

   364→/**
   365→ * Appel IA avec cascade fallback complète
   366→ * Ordre : Groq (Llama 3.3 70B, <10s) → Gemini Flash → Mistral-medium → Mistral-large → Workers AI
   367→ * Groq est gratuit et extrêmement rapide — parfait pour le pipeline CoT
   368→ */
   369→export async function callAI(env, systemPrompt, userPrompt, preferHighQuality = false, stageHint = 'default') {
   370→  const providers = [];
   371→  const maxTokens = MAX_TOKENS_BY_STAGE[stageHint] || MAX_TOKENS_BY_STAGE.default;
   372→
   373→  // 1. Groq — gratuit, Llama 3.3 70B, répond en <10s
   374→  if (env.GROQ_API_KEY) {
   375→    providers.push({ type: 'groq', model: 'llama-3.3-70b-versatile', key: env.GROQ_API_KEY, endpoint: GROQ_ENDPOINT });
   376→  }
   377→
   378→  // 2. Gemini Flash — gratuit, fiable, bon sur gros volumes
   379→  if (env.GEMINI_API_KEY) {
   380→    providers.push({ type: 'gemini', model: 'gemini-2.0-flash', key: env.GEMINI_API_KEY });
   381→  }
   382→
   383→  // 3. Mistral — premium, plus lent mais haute qualité
   384→  if (env.MISTRAL_API_KEY) {
   385→    // medium d'abord (plus rapide), large en backup
   386→    providers.push({ type: 'mistral', model: 'mistral-medium-latest', key: env.MISTRAL_API_KEY, endpoint: MISTRAL_ENDPOINT });
   387→    providers.push({ type: 'mistral', model: 'mistral-large-latest', key: env.MISTRAL_API_KEY, endpoint: MISTRAL_ENDPOINT });
   388→  }
   389→
   390→  // 4. Workers AI — dernier recours
   391→  providers.push({ type: 'workersai' });
   392→
   393→  const errors = [];
   394→
   395→  for (const provider of providers) {
   396→    const startTime = Date.now();
   397→    try {
   398→      let result;
   399→      if (provider.type === 'groq' || provider.type === 'mistral') {
   400→        result = await callOpenAICompatible(provider.endpoint, provider.model, provider.key, systemPrompt, userPrompt, maxTokens);
   401→      } else if (provider.type === 'gemini') {
   402→        result = await callGemini(provider.key, systemPrompt, userPrompt, maxTokens);
   403→      } else {
   404→        result = await callWorkersAI(env, `${systemPrompt}\n\n${userPrompt}`);
   405→      }
   406→      const duration = Date.now() - startTime;
   407→
   408→      // Vérifier que le contenu est substantiel
   409→      if (!result || result.trim().length < 50) {
   410→        const errMsg = `${provider.type}: réponse vide ou trop courte (${(result || '').length} chars)`;
   411→        errors.push(errMsg);
   412→        console.warn(`[callAI] ${errMsg}`);
   413→        continue;
   414→      }
   415→
   416→      console.log(`[callAI] ✅ ${provider.type}/${provider.model} — ${result.length} chars en ${(duration/1000).toFixed(1)}s`);
   417→      return { content: result, provider: provider.type, duration };
   418→    } catch (err) {
   419→      const duration = Date.now() - startTime;
   420→      const errMsg = `${provider.type}: ${err.message}`;
   421→      errors.push(errMsg);
   422→      console.error(`[callAI] ❌ ${provider.type}/${provider.model || ''} — ${err.message} (après ${(duration/1000).toFixed(1)}s)`);
   423→      continue;
   424→    }
   425→  }
   426→
   427→  throw new Error(`Tous les providers IA ont échoué: ${errors.join(' | ')}`);
   428→}

   658→// ============================================================
   659→// Fallback basé sur des règles (dernier recours absolu)
   660→// ============================================================
   661→function generateRuleBasedReview(articles) {
   662→  const lines = [];
   663→  lines.push('📌 **L\'ESSENTIEL DU JOUR**\n');
   664→  const headlines = articles.slice(0, 5);
   665→  for (const a of headlines) {
   666→    const desc = (a.description || '').replace(/<[^>]*>/g, '').substring(0, 120);
   667→    lines.push(`**${a.title}** — *${a.sourceName}*${desc ? ` : ${desc}` : ''}`);
   668→  }
   669→  lines.push('\n---\n');
   670→  lines.push('📰 **ANALYSE THÉMATIQUE**\n');
   671→  const grouped = {};
   672→  for (const a of articles) {
   673→    const cat = a.sourceCategory || 'Autre';
   674→    if (!grouped[cat]) grouped[cat] = [];
   675→    grouped[cat].push(a);
   676→  }
   677→  for (const [cat, items] of Object.entries(grouped)) {
   678→    lines.push(`**[${cat}]**`);
   679→    for (const a of items) {
   680→      const desc = (a.description || '').replace(/<[^>]*>/g, '').substring(0, 150);
   681→      lines.push(`- **${a.title}** (*${a.sourceName}*): ${desc}...`);
   682→    }
   683→    lines.push('');
   684→  }
   685→  lines.push('---\n');
   686→  lines.push('⚠️ *Mode dégradé — IA indisponible*');
   687→  return lines.join('\n');
   688→}
