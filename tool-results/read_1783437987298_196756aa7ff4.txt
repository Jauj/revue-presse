   241→   241→   241→/** Échappe les caractères XML dangereux pour éviter de casser le parsing */
   242→   242→   242→function escapeXML(str) {
   243→   243→   243→  return String(str)
   244→   244→   244→    .replace(/&/g, '&amp;')
   245→   245→   245→    .replace(/</g, '&lt;')
   246→   246→   246→    .replace(/>/g, '&gt;')
   247→   247→   247→    .replace(/"/g, '&quot;')
   248→   248→   248→    .replace(/'/g, '&apos;');
   249→   249→   249→}

   251→   251→   251→function formatDateForAI(dateStr) {
   252→   252→   252→  if (!dateStr) return 'date inconnue';
   253→   253→   253→  try {
   254→   254→   254→    const d = new Date(dateStr);
   255→   255→   255→    if (isNaN(d.getTime())) return dateStr;
   256→   256→   256→    return d.toLocaleDateString('fr-FR', { day: 'numeric', month: 'long', year: 'numeric', timeZone: 'Europe/Paris' });
   257→   257→   257→  } catch { return dateStr; }
   258→   258→   258→}

   260→   260→   260→function buildArticlesXML(articles, maxWordsPerArticle = 200) {
   261→   261→   261→  let xml = `<articles count="${articles.length}">\n`;
   262→   262→   262→  for (let i = 0; i < articles.length; i++) {
   263→   263→   263→    const a = articles[i];
   264→   264→   264→    const text = a.extractedText || a.description || '(indisponible)';
   265→   265→   265→    const excerpt = text.split(/\s+/).slice(0, maxWordsPerArticle).join(' ');
   266→   266→   266→    xml += `<article id="${i + 1}">\n`;
   267→   267→   267→    xml += `<source>${escapeXML(a.sourceName)}</source>\n`;
   268→   268→   268→    xml += `<lang>${escapeXML(a.sourceLang || 'fr')}</lang>\n`;
   269→   269→   269→    xml += `<category>${escapeXML(a.sourceCategory || 'autre')}</category>\n`;
   270→   270→   270→    xml += `<title>${escapeXML(a.title)}</title>\n`;
   271→   271→   271→    xml += `<pub_date>${escapeXML(formatDateForAI(a.pubDate))}</pub_date>\n`;
   272→   272→   272→    xml += `<content>\n${escapeXML(excerpt)}\n</content>\n`;
   273→   273→   273→    if (a.link) xml += `<url>${escapeXML(a.link)}</url>\n`;
   274→   274→   274→    xml += `</article>\n\n`;
   275→   275→   275→  }
   276→   276→   276→  xml += `</articles>`;
   277→   277→   277→  return xml;
   278→   278→   278→}

   306→   306→   306→async function callOpenAICompatible(endpoint, model, apiKey, systemPrompt, userPrompt, maxTokens = 8000) {
   307→   307→   307→  const response = await fetch(endpoint, {
   308→   308→   308→    method: 'POST',
   309→   309→   309→    headers: { 'Authorization': `Bearer ${apiKey}`, 'Content-Type': 'application/json' },
   310→   310→   310→    body: JSON.stringify({
   311→   311→   311→      model,
   312→   312→   312→      messages: [
   313→   313→   313→        { role: 'system', content: systemPrompt },
   314→   314→   314→        { role: 'user', content: userPrompt },
   315→   315→   315→      ],
   316→   316→   316→      temperature: 0.2,
   317→   317→   317→      max_tokens: maxTokens,
   318→   318→   318→      top_p: 0.9,
   319→   319→   319→    }),
   320→   320→   320→    signal: AbortSignal.timeout(180000),
   321→   321→   321→  });
   322→   322→   322→
   323→   323→   323→  if (!response.ok) {
   324→   324→   324→    const errBody = await response.text().catch(() => '');
   325→   325→   325→    throw new Error(`HTTP ${response.status}: ${errBody.substring(0, 200)}`);
   326→   326→   326→  }
   327→   327→   327→  return (await response.json()).choices[0].message.content;
   328→   328→   328→}
