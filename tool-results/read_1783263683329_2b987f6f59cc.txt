    30→    30→export async function saveDailyMemory(env, date, reviewContent, articles) {
    31→    31→  const dateStr = date.toISOString().split('T')[0];
    32→    32→  const key = `${MEM_PREFIX}day:${dateStr}`;
    33→    33→
    34→    34→  const themes = extractThemesFromReview(reviewContent);
    35→    35→  const sources = [...new Set(articles.map(a => a.sourceName))];
    36→    36→  const keyFacts = extractKeyFacts(reviewContent);
    37→    37→  const actors = extractActors(reviewContent);
    38→    38→  const entities = extractEntities(reviewContent);
    39→    39→
    40→    40→  const memory = {
    41→    41→    date: dateStr,
    42→    42→    themes,
    43→    43→    sources,
    44→    44→    keyFacts,
    45→    45→    actors,
    46→    46→    entities,
    47→    47→    articleCount: articles.length,
    48→    48→    sourceCount: sources.length,
    49→    49→    wordEstimate: reviewContent.split(/\s+/).length,
    50→    50→    createdAt: new Date().toISOString(),
    51→    51→  };
    52→    52→
    53→    53→  await env.CACHE.put(key, JSON.stringify(memory), { expirationTtl: 30 * 24 * 3600 });
    54→    54→
    55→    55→  // Mettre à jour l'index des jours
    56→    56→  await updateDayIndex(env, dateStr);
    57→    57→
    58→    58→  return memory;

    66→    66→export async function updateSemanticMemory(env, dailyMemory) {
    67→    67→  const key = `${MEM_PREFIX}themes`;
    68→    68→  const raw = await env.CACHE.get(key);
    69→    69→  let semantic = raw ? JSON.parse(raw) : { themes: {}, lastUpdated: null, totalDays: 0 };
    70→    70→
    71→    71→  for (const theme of dailyMemory.themes) {
    72→    72→    const normalized = normalizeTheme(theme);
    73→    73→    if (!semantic.themes[normalized]) {
    74→    74→      semantic.themes[normalized] = {
    75→    75→        name: theme,
    76→    76→        count: 0,
    77→    77→        firstSeen: dailyMemory.date,
    78→    78→        lastSeen: dailyMemory.date,
    79→    79→        relatedSources: [],
    80→    80→        relatedFacts: [],
    81→    81→        days: [],
    82→    82→      };
    83→    83→    }
    84→    84→    semantic.themes[normalized].count++;
    85→    85→    semantic.themes[normalized].lastSeen = dailyMemory.date;
    86→    86→    if (!semantic.themes[normalized].days.includes(dailyMemory.date)) {
    87→    87→      semantic.themes[normalized].days.push(dailyMemory.date);
    88→    88→    }
    89→    89→
    90→    90→    const existing = new Set(semantic.themes[normalized].relatedSources);
    91→    91→    for (const s of dailyMemory.sources) existing.add(s);
    92→    92→    semantic.themes[normalized].relatedSources = [...existing].slice(0, 10);
    93→    93→
    94→    94→    for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
    95→    95→      if (!semantic.themes[normalized].relatedFacts.includes(fact)) {
    96→    96→        semantic.themes[normalized].relatedFacts.push(fact);
    97→    97→      }
    98→    98→    }
    99→    99→    semantic.themes[normalized].relatedFacts =
   100→   100→      semantic.themes[normalized].relatedFacts.slice(-5);
   101→   101→  }
   102→   102→
   103→   103→  semantic.totalDays++;
   104→   104→  semantic.lastUpdated = new Date().toISOString();
   105→   105→
   106→   106→  await env.CACHE.put(key, JSON.stringify(semantic), { expirationTtl: 365 * 24 * 3600 });
   107→   107→  return semantic;
   108→   108→}

   111→   111→// COUCHE 3 — NARRATIVES (fil rouge inter-jours)
   112→   112→// KV: memory:narratives  (TTL 90j)
   113→   113→// Détecte les récits qui traversent plusieurs jours

   115→   115→
   116→   116→export async function updateNarratives(env, dailyMemory) {
   117→   117→  const key = `${MEM_PREFIX}narratives`;
   118→   118→  const raw = await env.CACHE.get(key);
   119→   119→  let narratives = raw ? JSON.parse(raw) : { active: [], archived: [], lastUpdated: null };
   120→   120→
   121→   121→  const todayThemes = dailyMemory.themes.map(t => t.toLowerCase());
   122→   122→  let touchedNarratives = [];
   123→   123→
   124→   124→  // 1. Essayer de rattacher les thèmes du jour à des narratives existantes
   125→   125→  for (const narrative of narratives.active) {
   126→   126→    let matched = false;
   127→   127→    for (const theme of todayThemes) {
   128→   128→      const similarity = computeThemeSimilarity(theme, narrative.title.toLowerCase());
   129→   129→      if (similarity > 0.4) {
   130→   130→        // Mettre à jour la narrative
   131→   131→        narrative.lastSeen = dailyMemory.date;
   132→   132→        narrative.dayCount = (narrative.dayCount || 1) + 1;
   133→   133→        narrative.days = narrative.days || [];
   134→   134→        if (!narrative.days.includes(dailyMemory.date)) {
   135→   135→          narrative.days.push(dailyMemory.date);
   136→   136→        }
   137→   137→        // Ajouter les nouveaux faits
   138→   138→        for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
   139→   139→          if (!narrative.facts.some(f => f.fact === fact)) {
   140→   140→            narrative.facts.push({ fact, date: dailyMemory.date });
   141→   141→          }
   142→   142→        }
   143→   143→        // Ajouter les sources
   144→   144→        const existingSources = new Set(narrative.sources || []);
   145→   145→        for (const s of dailyMemory.sources) existingSources.add(s);
   146→   146→        narrative.sources = [...existingSources];
   147→   147→        // Dernière évolution
   148→   148→        narrative.lastEvolution = dailyMemory.themes[0] || '';
   149→   149→        matched = true;
   150→   150→        touchedNarratives.push(narrative.id);
   151→   151→        break;
   152→   152→      }
   153→   153→    }
   154→   154→  }
   155→   155→
   156→   156→  // 2. Créer de nouvelles narratives pour les thèmes non rattachés
   157→   157→  for (const theme of dailyMemory.themes.slice(0, 3)) {
   158→   158→    const themeLower = theme.toLowerCase();
   159→   159→    const alreadyTracked = touchedNarratives.some(id => {
   160→   160→      const n = narratives.active.find(a => a.id === id);
   161→   161→      return n && computeThemeSimilarity(themeLower, n.title.toLowerCase()) > 0.5;
   162→   162→    });
   163→   163→    if (!alreadyTracked) {
   164→   164→      narratives.active.push({
   165→   165→        id: `nar_${dailyMemory.date}_${normalizeTheme(theme).substring(0, 20)}`,
   166→   166→        title: theme,
   167→   167→        firstSeen: dailyMemory.date,
   168→   168→        lastSeen: dailyMemory.date,
   169→   169→        dayCount: 1,
   170→   170→        days: [dailyMemory.date],
   171→   171→        facts: dailyMemory.keyFacts.slice(0, 2).map(f => ({ fact: f, date: dailyMemory.date })),
   172→   172→        sources: dailyMemory.sources.slice(0, 5),
   173→   173→        lastEvolution: theme,
   174→   174→        status: 'active',
   175→   175→      });
   176→   176→    }
   177→   177→  }
   178→   178→
   179→   179→  // 3. Archiver les narratives inactives (>7 jours sans update)
   180→   180→  narratives.active = narratives.active.filter(n => {
   181→   181→    const daysSince = daysBetween(n.lastSeen, dailyMemory.date);
   182→   182→    if (daysSince > 7 && n.dayCount < 3) return false; // supprimer les éphémères
   183→   183→    if (daysSince > 14) {
   184→   184→      n.status = 'archived';
   185→   185→      n.archivedAt = dailyMemory.date;
   186→   186→      narratives.archived = narratives.archived || [];
   187→   187→      narratives.archived.push(n);
   188→   188→      return false;
   189→   189→    }
   190→   190→    return true;
   191→   191→  });
   192→   192→
   193→   193→  // Garder max 10 narratives actives (les plus récentes)
   194→   194→  narratives.active.sort((a, b) => b.lastSeen.localeCompare(a.lastSeen));
   195→   195→  narratives.active = narratives.active.slice(0, 10);
   196→   196→
   197→   197→  // Garder max 20 narratives archivées
   198→   198→  if (narratives.archived) {
   199→   199→    narratives.archived = narratives.archived.slice(-20);
   200→   200→  }
   201→   201→
   202→   202→  narratives.lastUpdated = new Date().toISOString();
   203→   203→
   204→   204→  await env.CACHE.put(key, JSON.stringify(narratives), { expirationTtl: 90 * 24 * 3600 });
   205→   205→  return narratives;
   206→   206→}

   210→   210→// KV: memory:distilled:2026-07  (TTL 365j)

   215→   215→/**
   216→   216→ * Processus de rêve complet (distillation IA)
   217→   217→ * Lit les mémoires des derniers N jours et produit des insights
   218→   218→ */
   219→   219→export async function dreamDistill(env, daysBack = 14, force = false) {
   220→   220→  const today = new Date();
   221→   221→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   222→   222→  const distilledKey = `${MEM_PREFIX}distilled:${monthKey}`;
   223→   223→
   224→   224→  // Vérifier si on a déjà distillé récemment (pas plus d'une fois par 3 jours)
   225→   225→  if (!force) {
   226→   226→    const lastDreamRaw = await env.CACHE.get(`${MEM_PREFIX}last_dream`);
   227→   227→    if (lastDreamRaw) {
   228→   228→      const lastDream = JSON.parse(lastDreamRaw);
   229→   229→      const daysSinceDream = daysBetween(lastDream.date, today.toISOString().split('T')[0]);
   230→   230→      if (daysSinceDream < 3) {
   231→   231→        return { skipped: true, reason: `Dernier rêve il y a ${daysSinceDream}j (min 3)`, lastDream };
   232→   232→      }
   233→   233→    }
   234→   234→  }
   235→   235→
   236→   236→  // 1. Collecter les mémoires épisodiques
   237→   237→  const dailyMemories = [];
   238→   238→  for (let i = 1; i <= daysBack; i++) {
   239→   239→    const d = new Date(today);
   240→   240→    d.setDate(d.getDate() - i);
   241→   241→    const dateStr = d.toISOString().split('T')[0];
   242→   242→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   243→   243→    if (raw) dailyMemories.push(JSON.parse(raw));
   244→   244→  }
   245→   245→
   246→   246→  if (dailyMemories.length < 3) {
   247→   247→    return { skipped: true, reason: `Seulement ${dailyMemories.length} mémoires (min 3)`, hasEnough: false };
   248→   248→  }
   249→   249→
   250→   250→  // 2. Collecter les narratives
   251→   251→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   252→   252→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   253→   253→
   254→   254→  // 3. Construire le matériel de rêve (compact)
   255→   255→  const dreamMaterial = buildDreamMaterial(dailyMemories, narratives);
   256→   256→
   257→   257→  // 4. Appel IA de distillation
   258→   258→  const startTime = Date.now();
   259→   259→  let distilled = null;
   260→   260→  try {
   261→   261→    distilled = await callAIDream(env, dreamMaterial);
   262→   262→  } catch (err) {
   263→   263→    console.error(`Rêve IA échoué: ${err.message}`);
   264→   264→    // Fallback : distillation rule-based
   265→   265→    distilled = ruleBasedDistillation(dailyMemories, narratives);
   266→   266→  }
   267→   267→  const dreamDuration = Date.now() - startTime;
   268→   268→
   269→   269→  // 5. Stocker la distillation mensuelle
   270→   270→  const distilledData = {
   271→   271→    month: monthKey,
   272→   272→    date: today.toISOString().split('T')[0],
   273→   273→    daysAnalyzed: dailyMemories.length,
   274→   274→    dreamDurationMs: dreamDuration,
   275→   275→    aiPowered: !!distilled.aiGenerated,
   276→   276→    ...distilled,
   277→   277→  };
   278→   278→
   279→   279→  await env.CACHE.put(distilledKey, JSON.stringify(distilledData), { expirationTtl: 365 * 24 * 3600 });
   280→   280→
   281→   281→  // 6. Mettre à jour la mémoire éditoriale
   282→   282→  await updateEditorialMemory(env, dailyMemories, distilled);
   283→   283→
   284→   284→  // 7. Enregistrer le dernier rêve
   285→   285→  await env.CACHE.put(`${MEM_PREFIX}last_dream`, JSON.stringify({
   286→   286→    date: today.toISOString().split('T')[0],
   287→   287→    type: dailyMemories.length >= 7 ? 'deep' : 'light',
   288→   288→    duration: dreamDuration,
   289→   289→    daysAnalyzed: dailyMemories.length,
   290→   290→    monthKey,
   291→   291→  }), { expirationTtl: 30 * 24 * 3600 });
   292→   292→
   293→   293→  // 8. Nettoyer les vieilles mémoires épisodiques (au-delà de 14 jours)
   294→   294→  await pruneOldMemories(env, 14);
   295→   295→
   296→   296→  return {
   297→   297→    success: true,
   298→   298→    daysAnalyzed: dailyMemories.length,
   299→   299→    dreamDurationMs: dreamDuration,
   300→   300→    monthKey,
   301→   301→    distilledThemes: distilled.patterns?.length || 0,
   302→   302→    narrativeCount: narratives?.active?.length || 0,
   303→   303→  };
   304→   304→}

   307→   307→// COUCHE 5 — MÉMOIRE ÉDITORIALE (préférences apprises)
   308→   308→// KV: memory:editorial  (TTL 365j)
   309→   309→// Apprend quels patterns fonctionnent bien
   310→   310→// ============================================================
   311→   311→
   312→   312→async function updateEditorialMemory(env, dailyMemories, distilled) {
   313→   313→  const key = `${MEM_PREFIX}editorial`;
   314→   314→  const raw = await env.CACHE.get(key);
   315→   315→  let editorial = raw ? JSON.parse(raw) : {
   316→   316→    sourcePatterns: {},
   317→   317→    topicBalance: {},
   318→   318→    avgArticleCount: 0,
   319→   319→    avgThemesPerDay: 0,
   320→   320→    totalReviews: 0,
   321→   321→    preferences: [],
   322→   322→    lastUpdated: null,
   323→   323→  };
   324→   324→
   325→   325→  // Statistiques glissantes
   326→   326→  editorial.totalReviews++;
   327→   327→  const recentCounts = dailyMemories.slice(-7).map(m => m.articleCount);
   328→   328→  editorial.avgArticleCount = Math.round(recentCounts.reduce((a, b) => a + b, 0) / recentCounts.length);
   329→   329→  const recentThemes = dailyMemories.slice(-7).map(m => m.themes.length);
   330→   330→  editorial.avgThemesPerDay = Math.round(recentThemes.reduce((a, b) => a + b, 0) / recentThemes.length);
   331→   331→
   332→   332→  // Patterns par source
   333→   333→  for (const mem of dailyMemories) {
   334→   334→    for (const source of mem.sources) {
   335→   335→      if (!editorial.sourcePatterns[source]) {
   336→   336→        editorial.sourcePatterns[source] = { appearances: 0, themesContributed: [] };
   337→   337→      }
   338→   338→      editorial.sourcePatterns[source].appearances++;
   339→   339→      for (const theme of mem.themes.slice(0, 2)) {
   340→   340→        const existing = new Set(editorial.sourcePatterns[source].themesContributed);
   341→   341→        existing.add(theme);
   342→   342→        editorial.sourcePatterns[source].themesContributed = [...existing].slice(0, 10);
   343→   343→      }
   344→   344→    }
   345→   345→  }
   346→   346→
   347→   347→  // Préférences apprises (depuis la distillation IA)
   348→   348→  if (distilled.insights) {
   349→   349→    for (const insight of distilled.insights.slice(0, 5)) {
   350→   350→      if (!editorial.preferences.some(p => p.insight === insight)) {
   351→   351→        editorial.preferences.push({
   352→   352→          insight,
   353→   353→          learnedFrom: dailyMemories[dailyMemories.length - 1]?.date,
   354→   354→          createdAt: new Date().toISOString(),
   355→   355→        });
   356→   356→      }
   357→   357→    }
   358→   358→    editorial.preferences = editorial.preferences.slice(-10);
   359→   359→  }
   360→   360→
   361→   361→  editorial.lastUpdated = new Date().toISOString();
   362→   362→  await env.CACHE.put(key, JSON.stringify(editorial), { expirationTtl: 365 * 24 * 3600 });
   363→   363→  return editorial;
   364→   364→}

   369→   369→
   370→   370→export async function getMemoryContext(env, maxDays = 7) {
   371→   371→  const today = new Date();
   372→   372→  const dailyMemories = [];
   373→   373→  const themesRaw = await env.CACHE.get(`${MEM_PREFIX}themes`);
   374→   374→  const semantic = themesRaw ? JSON.parse(themesRaw) : null;
   375→   375→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   376→   376→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   377→   377→  const editorialRaw = await env.CACHE.get(`${MEM_PREFIX}editorial`);
   378→   378→  const editorial = editorialRaw ? JSON.parse(editorialRaw) : null;
   379→   379→
   380→   380→  // Dernière distillation
   381→   381→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   382→   382→  const lastMonthKey = today.getMonth() === 0
   383→   383→    ? `${today.getFullYear() - 1}-12`
   384→   384→    : `${today.getFullYear()}-${String(today.getMonth()).padStart(2, '0')}`;
   385→   385→  const distilledRaw = await env.CACHE.get(`${MEM_PREFIX}distilled:${monthKey}`)
   386→   386→    || await env.CACHE.get(`${MEM_PREFIX}distilled:${lastMonthKey}`);
   387→   387→  const distilled = distilledRaw ? JSON.parse(distilledRaw) : null;
   388→   388→
   389→   389→  // Lire les derniers jours de mémoire épisodique
   390→   390→  for (let i = 1; i <= maxDays; i++) {
   391→   391→    const d = new Date(today);
   392→   392→    d.setDate(d.getDate() - i);
   393→   393→    const dateStr = d.toISOString().split('T')[0];
   394→   394→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   395→   395→    if (raw) dailyMemories.push(JSON.parse(raw));
   396→   396→  }
   397→   397→
   398→   398→  if (dailyMemories.length === 0 && !semantic && !distilled) {
   399→   399→    return null;
   400→   400→  }
   401→   401→
   402→   402→  return buildRichContextString(dailyMemories, semantic, narratives, distilled, editorial);
   403→   403→}

   406→   406→// NETTOYAGE — Supprimer les anciennes mémoires
   407→   407→// ============================================================
   408→   408→
   409→   409→export async function pruneOldMemories(env, keepDays = 14) {
   410→   410→  const today = new Date();
   411→   411→  let pruned = 0;
   412→   412→
   413→   413→  for (let i = keepDays + 1; i <= 60; i++) {
   414→   414→    const d = new Date(today);
   415→   415→    d.setDate(d.getDate() - i);
   416→   416→    const dateStr = d.toISOString().split('T')[0];
   417→   417→    const key = `${MEM_PREFIX}day:${dateStr}`;
   418→   418→    const existing = await env.CACHE.get(key);
   419→   419→    if (existing) {
   420→   420→      await env.CACHE.delete(key);
   421→   421→      pruned++;
   422→   422→    }
   423→   423→  }
   424→   424→
   425→   425→  return pruned;
   426→   426→}

   660→   660→function ruleBasedDistillation(dailyMemories, narratives) {
   661→   661→  // Patterns récurrents
   662→   662→  const themeFreq = {};
   663→   663→  for (const mem of dailyMemories) {
   664→   664→    for (const t of mem.themes) {
   665→   665→      const norm = normalizeTheme(t);
   666→   666→      themeFreq[norm] = (themeFreq[norm] || 0) + 1;
   667→   667→    }
   668→   668→  }
   669→   669→
   670→   670→  const patterns = Object.entries(themeFreq)
   671→   671→    .sort((a, b) => b[1] - a[1])
   672→   672→    .slice(0, 6)
   673→   673→    .map(([key, count]) => ({
   674→   674→      type: 'recurring',
   675→   675→      theme: key.replace(/_/g, ' '),
   676→   676→      frequency: `${count}/${dailyMemories.length} jours`,
   677→   677→      trend: count > dailyMemories.length * 0.5 ? 'stable' : 'emerging',
   678→   678→      note: `Présent ${count} jours sur ${dailyMemories.length}`,
   679→   679→    }));
   680→   680→
   681→   681→  // Narratives depuis la mémoire narratives
   682→   682→  const narList = (narratives?.active || []).slice(0, 4).map(n => ({
   683→   683→    title: n.title,
   684→   684→    arc: `${n.dayCount} jours d'évolution`,
   685→   685→    key_moments: n.facts?.slice(-2).map(f => f.fact) || [],
   686→   686→    status: n.dayCount >= 5 ? 'peaked' : 'active',
   687→   687→  }));
   688→   688→
   689→   689→  // Source dynamics
   690→   690→  const sourceFreq = {};
   691→   691→  for (const mem of dailyMemories) {
   692→   692→    for (const s of mem.sources) {
   693→   693→      sourceFreq[s] = (sourceFreq[s] || 0) + 1;
   694→   694→    }
   695→   695→  }
   696→   696→  const sourceDynamics = Object.entries(sourceFreq)
   697→   697→    .sort((a, b) => b[1] - a[1])
   698→   698→    .slice(0, 5)
   699→   699→    .map(([source, count]) => ({
   700→   700→      source,
   701→   701→      role: count > dailyMemories.length * 0.7 ? 'leader' : 'contributor',
   702→   702→      notes: `Présent ${count} jours`,
   703→   703→    }));
   704→   704→
   705→   705→  return {
   706→   706→    patterns,
   707→   707→    narratives: narList,
   708→   708→    insights: [],
   709→   709→    source_dynamics: sourceDynamics,
   710→   710→    blind_spots: [],
   711→   711→    macro_trends: [],
   712→   712→    aiGenerated: false,
   713→   713→    generatedAt: new Date().toISOString(),
   714→   714→  };
   715→   715→}

   816→   816→
   817→   817→function extractThemesFromReview(reviewText) {
   818→   818→  const themes = [];
   819→   819→  const nonThemes = new Set([
   820→   820→    "L'ESSENTIEL DU JOUR", "ANALYSE THÉMATIQUE", "TENDANCES",
   821→   821→    "CHIFFRES CLÉS", "À SURVEILLER", "ESSENTIEL", "TENDANCES & PERSPECTIVES",
   822→   822→    "SOMMAIRE", "POINTS DE TENSION", "SOURCES", "POUR ALLER PLUS LOIN",
   823→   823→    "APPROFONDIR", "NOTE FINALE",
   824→   824→  ]);
   825→   825→
   826→   826→  // Format éditorial : **1. Titre éditorial** ou **1. Titre : sous-titre**
   827→   827→  const editorialRegex = /^\*\*(\d+)\.[\s]+(.+?)(?:\s*[:：]\s*(.+?))?\*\*$/gm;
   828→   828→  let match;
   829→   829→  while ((match = editorialRegex.exec(reviewText)) !== null) {
   830→   830→    const title = match[2].trim();
   831→   831→    if (title.length > 5 && !nonThemes.has(title.toUpperCase())) {
   832→   832→      themes.push(title);
   833→   833→    }
   834→   834→  }
   835→   835→
   836→   836→  // Format Smart Brevity legacy : **THÈME 1** suivi de :
   837→   837→  const sectionRegex = /\*\*([^*\d]{3,40})\*\*\s*[:：]/g;
   838→   838→  while ((match = sectionRegex.exec(reviewText)) !== null) {
   839→   839→    const theme = match[1].trim();
   840→   840→    if (!nonThemes.has(theme.toUpperCase()) && !themes.includes(theme)) {
   841→   841→      themes.push(theme);
   842→   842→    }
   843→   843→  }
   844→   844→
   845→   845→  // Format axiom legacy : **Bold Axiom :**
   846→   846→  const axiomRegex = /\*\*([^*]{2,25})\s*[:：]\*\*/g;
   847→   847→  while ((match = axiomRegex.exec(reviewText)) !== null) {
   848→   848→    const axiom = match[1].trim();
   849→   849→    if (axiom.length > 3 && axiom.length < 25 && !nonThemes.has(axiom.toUpperCase()) && !themes.includes(axiom)) {
   850→   850→      themes.push(axiom);
   851→   851→    }

   857→   857→function extractKeyFacts(reviewText) {
   858→   858→  const facts = [];
   859→   859→  const seen = new Set();
   860→   860→
   861→   861→  // 1. Lignes à puces substantielles (30-250 chars)
   862→   862→  const bulletRegex = /^-\s+(.{30,250})$/gm;
   863→   863→  let match;
   864→   864→  while ((match = bulletRegex.exec(reviewText)) !== null) {
   865→   865→    const fact = match[1].trim();
   866→   866→    // Exclure les lignes qui sont juste des titres de sections ou des liens
   867→   867→    if (!fact.startsWith('**') && !fact.startsWith('http') && !seen.has(fact.substring(0, 50))) {
   868→   868→      facts.push(fact);
   869→   869→      seen.add(fact.substring(0, 50));
   870→   870→    }
   871→   871→  }
   872→   872→
   873→   873→  // 2. Phrases avec des chiffres précis dans les paragraphes (signal de faits durs)
   874→   874→  const numberRegex = /[^.!?]*(?:\d{2,}|\d+\s*(?:millions?|milliards?|%|morts?|blessés?|personnes|jours|ans))[^.!?]*[.!?]/gi;
   875→   875→  while ((match = numberRegex.exec(reviewText)) !== null) {
   876→   876→    const fact = match[0].trim();
   877→   877→    if (fact.length > 30 && fact.length < 250 && !seen.has(fact.substring(0, 50))) {
   878→   878→      facts.push(fact);
   879→   879→      seen.add(fact.substring(0, 50));
   880→   880→    }
   881→   881→  }
   882→   882→
   883→   883→  return [...new Set(facts)].slice(0, 15);
   884→   884→}

  1019→  1019→function computeThemeSimilarity(a, b) {
  1020→  1020→  const wordsA = new Set(a.split(/[\s_]+/));
  1021→  1021→  const wordsB = new Set(b.split(/[\s_]+/));
  1022→  1022→  if (wordsA.size === 0 || wordsB.size === 0) return 0;
  1023→  1023→  let intersection = 0;
  1024→  1024→  for (const w of wordsA) {
  1025→  1025→    if (wordsB.has(w)) intersection++;
  1026→  1026→  }
  1027→  1027→  return intersection / Math.max(wordsA.size, wordsB.size);
  1028→  1028→}
