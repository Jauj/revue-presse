    30→export async function saveDailyMemory(env, date, reviewContent, articles) {
    31→  const dateStr = date.toISOString().split('T')[0];
    32→  const key = `${MEM_PREFIX}day:${dateStr}`;
    33→
    34→  const themes = extractThemesFromReview(reviewContent);
    35→  const sources = [...new Set(articles.map(a => a.sourceName))];
    36→  const keyFacts = extractKeyFacts(reviewContent);
    37→  const actors = extractActors(reviewContent);
    38→  const entities = extractEntities(reviewContent);
    39→
    40→  const memory = {
    41→    date: dateStr,
    42→    themes,
    43→    sources,
    44→    keyFacts,
    45→    actors,
    46→    entities,
    47→    articleCount: articles.length,
    48→    sourceCount: sources.length,
    49→    wordEstimate: reviewContent.split(/\s+/).length,
    50→    createdAt: new Date().toISOString(),
    51→  };
    52→
    53→  await env.CACHE.put(key, JSON.stringify(memory), { expirationTtl: 30 * 24 * 3600 });
    54→
    55→  // Mettre à jour l'index des jours
    56→  await updateDayIndex(env, dateStr);
    57→
    58→  return memory;
    59→}

    61→// ============================================================
    62→// COUCHE 2 — MÉMOIRE SÉMANTIQUE (thèmes récurrents)
    63→// KV: memory:themes  (TTL 365j)
    64→// ============================================================
    65→
    66→export async function updateSemanticMemory(env, dailyMemory) {
    67→  const key = `${MEM_PREFIX}themes`;
    68→  const raw = await env.CACHE.get(key);
    69→  let semantic = raw ? JSON.parse(raw) : { themes: {}, lastUpdated: null, totalDays: 0 };
    70→
    71→  for (const theme of dailyMemory.themes) {
    72→    const normalized = normalizeTheme(theme);
    73→    if (!semantic.themes[normalized]) {
    74→      semantic.themes[normalized] = {
    75→        name: theme,
    76→        count: 0,
    77→        firstSeen: dailyMemory.date,
    78→        lastSeen: dailyMemory.date,
    79→        relatedSources: [],
    80→        relatedFacts: [],
    81→        days: [],
    82→      };
    83→    }
    84→    semantic.themes[normalized].count++;
    85→    semantic.themes[normalized].lastSeen = dailyMemory.date;
    86→    if (!semantic.themes[normalized].days.includes(dailyMemory.date)) {
    87→      semantic.themes[normalized].days.push(dailyMemory.date);
    88→    }
    89→
    90→    const existing = new Set(semantic.themes[normalized].relatedSources);
    91→    for (const s of dailyMemory.sources) existing.add(s);
    92→    semantic.themes[normalized].relatedSources = [...existing].slice(0, 10);
    93→
    94→    for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
    95→      if (!semantic.themes[normalized].relatedFacts.includes(fact)) {
    96→        semantic.themes[normalized].relatedFacts.push(fact);
    97→      }
    98→    }
    99→    semantic.themes[normalized].relatedFacts =
   100→      semantic.themes[normalized].relatedFacts.slice(-5);
   101→  }
   102→
   103→  semantic.totalDays++;
   104→  semantic.lastUpdated = new Date().toISOString();
   105→
   106→  await env.CACHE.put(key, JSON.stringify(semantic), { expirationTtl: 365 * 24 * 3600 });
   107→  return semantic;
   108→}

   111→// COUCHE 3 — NARRATIVES (fil rouge inter-jours)
   112→// KV: memory:narratives  (TTL 90j)
   113→// Détecte les récits qui traversent plusieurs jours

   115→
   116→export async function updateNarratives(env, dailyMemory) {
   117→  const key = `${MEM_PREFIX}narratives`;
   118→  const raw = await env.CACHE.get(key);
   119→  let narratives = raw ? JSON.parse(raw) : { active: [], archived: [], lastUpdated: null };
   120→
   121→  const todayThemes = dailyMemory.themes.map(t => t.toLowerCase());
   122→  let touchedNarratives = [];
   123→
   124→  // 1. Essayer de rattacher les thèmes du jour à des narratives existantes
   125→  for (const narrative of narratives.active) {
   126→    let matched = false;
   127→    for (const theme of todayThemes) {
   128→      const similarity = computeThemeSimilarity(theme, narrative.title.toLowerCase());
   129→      if (similarity > 0.4) {
   130→        // Mettre à jour la narrative
   131→        narrative.lastSeen = dailyMemory.date;
   132→        narrative.dayCount = (narrative.dayCount || 1) + 1;
   133→        narrative.days = narrative.days || [];
   134→        if (!narrative.days.includes(dailyMemory.date)) {
   135→          narrative.days.push(dailyMemory.date);
   136→        }
   137→        // Ajouter les nouveaux faits
   138→        for (const fact of dailyMemory.keyFacts.slice(0, 2)) {
   139→          if (!narrative.facts.some(f => f.fact === fact)) {
   140→            narrative.facts.push({ fact, date: dailyMemory.date });
   141→          }
   142→        }
   143→        // Ajouter les sources
   144→        const existingSources = new Set(narrative.sources || []);
   145→        for (const s of dailyMemory.sources) existingSources.add(s);
   146→        narrative.sources = [...existingSources];
   147→        // Dernière évolution
   148→        narrative.lastEvolution = dailyMemory.themes[0] || '';
   149→        matched = true;
   150→        touchedNarratives.push(narrative.id);
   151→        break;
   152→      }
   153→    }
   154→  }
   155→
   156→  // 2. Créer de nouvelles narratives pour les thèmes non rattachés
   157→  for (const theme of dailyMemory.themes.slice(0, 3)) {
   158→    const themeLower = theme.toLowerCase();
   159→    const alreadyTracked = touchedNarratives.some(id => {
   160→      const n = narratives.active.find(a => a.id === id);
   161→      return n && computeThemeSimilarity(themeLower, n.title.toLowerCase()) > 0.5;
   162→    });
   163→    if (!alreadyTracked) {
   164→      narratives.active.push({
   165→        id: `nar_${dailyMemory.date}_${normalizeTheme(theme).substring(0, 20)}`,
   166→        title: theme,
   167→        firstSeen: dailyMemory.date,
   168→        lastSeen: dailyMemory.date,
   169→        dayCount: 1,
   170→        days: [dailyMemory.date],
   171→        facts: dailyMemory.keyFacts.slice(0, 2).map(f => ({ fact: f, date: dailyMemory.date })),
   172→        sources: dailyMemory.sources.slice(0, 5),
   173→        lastEvolution: theme,
   174→        status: 'active',
   175→      });
   176→    }
   177→  }
   178→
   179→  // 3. Archiver les narratives inactives (>7 jours sans update)
   180→  narratives.active = narratives.active.filter(n => {
   181→    const daysSince = daysBetween(n.lastSeen, dailyMemory.date);
   182→    if (daysSince > 7 && n.dayCount < 3) return false; // supprimer les éphémères
   183→    if (daysSince > 14) {
   184→      n.status = 'archived';
   185→      n.archivedAt = dailyMemory.date;
   186→      narratives.archived = narratives.archived || [];
   187→      narratives.archived.push(n);
   188→      return false;
   189→    }
   190→    return true;
   191→  });
   192→
   193→  // Garder max 10 narratives actives (les plus récentes)
   194→  narratives.active.sort((a, b) => b.lastSeen.localeCompare(a.lastSeen));
   195→  narratives.active = narratives.active.slice(0, 10);
   196→
   197→  // Garder max 20 narratives archivées
   198→  if (narratives.archived) {
   199→    narratives.archived = narratives.archived.slice(-20);
   200→  }
   201→
   202→  narratives.lastUpdated = new Date().toISOString();
   203→
   204→  await env.CACHE.put(key, JSON.stringify(narratives), { expirationTtl: 90 * 24 * 3600 });
   205→  return narratives;
   206→}

   209→// COUCHE 4 — MÉMOIRE DISTILLÉE (connaissances consolidées par IA)
   210→// KV: memory:distilled:2026-07  (TTL 365j)
   211→// "Le rêve" : l'IA analyse N jours de mémoires et produit
   212→// un résumé compact de connaissances de haut niveau
   213→// ============================================================
   214→
   215→/**
   216→ * Processus de rêve complet (distillation IA)
   217→ * Lit les mémoires des derniers N jours et produit des insights
   218→ */
   219→export async function dreamDistill(env, daysBack = 14, force = false) {
   220→  const today = new Date();
   221→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   222→  const distilledKey = `${MEM_PREFIX}distilled:${monthKey}`;
   223→
   224→  // Vérifier si on a déjà distillé récemment (pas plus d'une fois par 3 jours)
   225→  if (!force) {
   226→    const lastDreamRaw = await env.CACHE.get(`${MEM_PREFIX}last_dream`);
   227→    if (lastDreamRaw) {
   228→      const lastDream = JSON.parse(lastDreamRaw);
   229→      const daysSinceDream = daysBetween(lastDream.date, today.toISOString().split('T')[0]);
   230→      if (daysSinceDream < 3) {
   231→        return { skipped: true, reason: `Dernier rêve il y a ${daysSinceDream}j (min 3)`, lastDream };
   232→      }
   233→    }
   234→  }
   235→
   236→  // 1. Collecter les mémoires épisodiques
   237→  const dailyMemories = [];
   238→  for (let i = 1; i <= daysBack; i++) {
   239→    const d = new Date(today);
   240→    d.setDate(d.getDate() - i);
   241→    const dateStr = d.toISOString().split('T')[0];
   242→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   243→    if (raw) dailyMemories.push(JSON.parse(raw));
   244→  }
   245→
   246→  if (dailyMemories.length < 3) {
   247→    return { skipped: true, reason: `Seulement ${dailyMemories.length} mémoires (min 3)`, hasEnough: false };
   248→  }
   249→
   250→  // 2. Collecter les narratives
   251→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   252→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   253→
   254→  // 3. Construire le matériel de rêve (compact)
   255→  const dreamMaterial = buildDreamMaterial(dailyMemories, narratives);
   256→
   257→  // 4. Appel IA de distillation
   258→  const startTime = Date.now();
   259→  let distilled = null;
   260→  try {
   261→    distilled = await callAIDream(env, dreamMaterial);
   262→  } catch (err) {
   263→    console.error(`Rêve IA échoué: ${err.message}`);
   264→    // Fallback : distillation rule-based
   265→    distilled = ruleBasedDistillation(dailyMemories, narratives);
   266→  }
   267→  const dreamDuration = Date.now() - startTime;
   268→
   269→  // 5. Stocker la distillation mensuelle
   270→  const distilledData = {
   271→    month: monthKey,
   272→    date: today.toISOString().split('T')[0],
   273→    daysAnalyzed: dailyMemories.length,
   274→    dreamDurationMs: dreamDuration,
   275→    aiPowered: !!distilled.aiGenerated,
   276→    ...distilled,
   277→  };
   278→
   279→  await env.CACHE.put(distilledKey, JSON.stringify(distilledData), { expirationTtl: 365 * 24 * 3600 });
   280→
   281→  // 6. Mettre à jour la mémoire éditoriale
   282→  await updateEditorialMemory(env, dailyMemories, distilled);
   283→
   284→  // 7. Enregistrer le dernier rêve
   285→  await env.CACHE.put(`${MEM_PREFIX}last_dream`, JSON.stringify({
   286→    date: today.toISOString().split('T')[0],
   287→    type: dailyMemories.length >= 7 ? 'deep' : 'light',
   288→    duration: dreamDuration,
   289→    daysAnalyzed: dailyMemories.length,
   290→    monthKey,
   291→  }), { expirationTtl: 30 * 24 * 3600 });
   292→
   293→  // 8. Nettoyer les vieilles mémoires épisodiques (au-delà de 14 jours)
   294→  await pruneOldMemories(env, 14);
   295→
   296→  return {
   297→    success: true,
   298→    daysAnalyzed: dailyMemories.length,
   299→    dreamDurationMs: dreamDuration,
   300→    monthKey,
   301→    distilledThemes: distilled.patterns?.length || 0,
   302→    narrativeCount: narratives?.active?.length || 0,
   303→  };
   304→}

   306→// ============================================================
   307→// COUCHE 5 — MÉMOIRE ÉDITORIALE (préférences apprises)
   308→// KV: memory:editorial  (TTL 365j)
   309→// Apprend quels patterns fonctionnent bien
   310→// ============================================================
   311→
   312→async function updateEditorialMemory(env, dailyMemories, distilled) {
   313→  const key = `${MEM_PREFIX}editorial`;
   314→  const raw = await env.CACHE.get(key);
   315→  let editorial = raw ? JSON.parse(raw) : {
   316→    sourcePatterns: {},
   317→    topicBalance: {},
   318→    avgArticleCount: 0,
   319→    avgThemesPerDay: 0,
   320→    totalReviews: 0,
   321→    preferences: [],
   322→    lastUpdated: null,
   323→  };
   324→
   325→  // Statistiques glissantes
   326→  editorial.totalReviews++;
   327→  const recentCounts = dailyMemories.slice(-7).map(m => m.articleCount);
   328→  editorial.avgArticleCount = Math.round(recentCounts.reduce((a, b) => a + b, 0) / recentCounts.length);
   329→  const recentThemes = dailyMemories.slice(-7).map(m => m.themes.length);
   330→  editorial.avgThemesPerDay = Math.round(recentThemes.reduce((a, b) => a + b, 0) / recentThemes.length);
   331→
   332→  // Patterns par source
   333→  for (const mem of dailyMemories) {
   334→    for (const source of mem.sources) {
   335→      if (!editorial.sourcePatterns[source]) {
   336→        editorial.sourcePatterns[source] = { appearances: 0, themesContributed: [] };
   337→      }
   338→      editorial.sourcePatterns[source].appearances++;
   339→      for (const theme of mem.themes.slice(0, 2)) {
   340→        const existing = new Set(editorial.sourcePatterns[source].themesContributed);
   341→        existing.add(theme);
   342→        editorial.sourcePatterns[source].themesContributed = [...existing].slice(0, 10);
   343→      }
   344→    }
   345→  }
   346→
   347→  // Préférences apprises (depuis la distillation IA)
   348→  if (distilled.insights) {
   349→    for (const insight of distilled.insights.slice(0, 5)) {
   350→      if (!editorial.preferences.some(p => p.insight === insight)) {
   351→        editorial.preferences.push({
   352→          insight,
   353→          learnedFrom: dailyMemories[dailyMemories.length - 1]?.date,
   354→          createdAt: new Date().toISOString(),
   355→        });
   356→      }
   357→    }
   358→    editorial.preferences = editorial.preferences.slice(-10);
   359→  }
   360→
   361→  editorial.lastUpdated = new Date().toISOString();
   362→  await env.CACHE.put(key, JSON.stringify(editorial), { expirationTtl: 365 * 24 * 3600 });
   363→  return editorial;
   364→}

   367→// LECTURE — Construire le contexte mémoire pour injection CoT
   368→// ============================================================
   369→
   370→export async function getMemoryContext(env, maxDays = 7) {
   371→  const today = new Date();
   372→  const dailyMemories = [];
   373→  const themesRaw = await env.CACHE.get(`${MEM_PREFIX}themes`);
   374→  const semantic = themesRaw ? JSON.parse(themesRaw) : null;
   375→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   376→  const narratives = narrativesRaw ? JSON.parse(narrativesRaw) : null;
   377→  const editorialRaw = await env.CACHE.get(`${MEM_PREFIX}editorial`);
   378→  const editorial = editorialRaw ? JSON.parse(editorialRaw) : null;
   379→
   380→  // Dernière distillation
   381→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   382→  const lastMonthKey = today.getMonth() === 0
   383→    ? `${today.getFullYear() - 1}-12`
   384→    : `${today.getFullYear()}-${String(today.getMonth()).padStart(2, '0')}`;
   385→  const distilledRaw = await env.CACHE.get(`${MEM_PREFIX}distilled:${monthKey}`)
   386→    || await env.CACHE.get(`${MEM_PREFIX}distilled:${lastMonthKey}`);
   387→  const distilled = distilledRaw ? JSON.parse(distilledRaw) : null;
   388→
   389→  // Lire les derniers jours de mémoire épisodique
   390→  for (let i = 1; i <= maxDays; i++) {
   391→    const d = new Date(today);
   392→    d.setDate(d.getDate() - i);
   393→    const dateStr = d.toISOString().split('T')[0];
   394→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   395→    if (raw) dailyMemories.push(JSON.parse(raw));
   396→  }
   397→
   398→  if (dailyMemories.length === 0 && !semantic && !distilled) {
   399→    return null;
   400→  }
   401→
   402→  return buildRichContextString(dailyMemories, semantic, narratives, distilled, editorial);
   403→}

   406→// NETTOYAGE — Supprimer les anciennes mémoires
   407→// ============================================================
   408→
   409→export async function pruneOldMemories(env, keepDays = 14) {
   410→  const today = new Date();
   411→  let pruned = 0;
   412→
   413→  for (let i = keepDays + 1; i <= 60; i++) {
   414→    const d = new Date(today);
   415→    d.setDate(d.getDate() - i);
   416→    const dateStr = d.toISOString().split('T')[0];
   417→    const key = `${MEM_PREFIX}day:${dateStr}`;
   418→    const existing = await env.CACHE.get(key);
   419→    if (existing) {
   420→      await env.CACHE.delete(key);
   421→      pruned++;
   422→    }
   423→  }
   424→
   425→  return pruned;
   426→}

   432→export async function getMemoryStats(env) {
   433→  const today = new Date();
   434→  const stats = {
   435→    totalDailyMemories: 0,
   436→    semanticThemes: 0,
   437→    activeNarratives: 0,
   438→    hasDistillation: false,
   439→    hasEditorial: false,
   440→    lastDream: null,
   441→    topThemes: [],
   442→    activeNarrativeTitles: [],
   443→  };
   444→
   445→  // Compter les mémoires quotidiennes (derniers 14 jours)
   446→  for (let i = 0; i <= 14; i++) {
   447→    const d = new Date(today);
   448→    d.setDate(d.getDate() - i);
   449→    const dateStr = d.toISOString().split('T')[0];
   450→    const raw = await env.CACHE.get(`${MEM_PREFIX}day:${dateStr}`);
   451→    if (raw) stats.totalDailyMemories++;
   452→  }
   453→
   454→  // Thèmes sémantiques
   455→  const themesRaw = await env.CACHE.get(`${MEM_PREFIX}themes`);
   456→  if (themesRaw) {
   457→    const semantic = JSON.parse(themesRaw);
   458→    stats.semanticThemes = Object.keys(semantic.themes).length;
   459→    stats.totalDays = semantic.totalDays;
   460→    stats.topThemes = Object.entries(semantic.themes)
   461→      .sort((a, b) => b[1].count - a[1].count)
   462→      .slice(0, 8)
   463→      .map(([k, v]) => ({ theme: v.name, count: v.count, lastSeen: v.lastSeen, days: v.days?.length || 0 }));
   464→  }
   465→
   466→  // Narratives
   467→  const narrativesRaw = await env.CACHE.get(`${MEM_PREFIX}narratives`);
   468→  if (narrativesRaw) {
   469→    const narratives = JSON.parse(narrativesRaw);
   470→    stats.activeNarratives = narratives.active?.length || 0;
   471→    stats.activeNarrativeTitles = (narratives.active || []).slice(0, 5).map(n => ({
   472→      title: n.title, days: n.dayCount, lastSeen: n.lastSeen,
   473→    }));
   474→  }
   475→
   476→  // Distillation
   477→  const monthKey = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;
   478→  const distilledRaw = await env.CACHE.get(`${MEM_PREFIX}distilled:${monthKey}`);
   479→  if (distilledRaw) {
   480→    stats.hasDistillation = true;
   481→    const d = JSON.parse(distilledRaw);
   482→    stats.distillationInfo = { month: d.month, daysAnalyzed: d.daysAnalyzed, aiPowered: d.aiPowered };
   483→  }
   484→
   485→  // Éditorial
   486→  const editorialRaw = await env.CACHE.get(`${MEM_PREFIX}editorial`);
   487→  if (editorialRaw) {
   488→    stats.hasEditorial = true;
   489→    const e = JSON.parse(editorialRaw);
   490→    stats.editorialInfo = { totalReviews: e.totalReviews, avgArticles: e.avgArticleCount, preferences: e.preferences?.length || 0 };
   491→  }
   492→
   493→  // Dernier rêve
   494→  const lastDreamRaw = await env.CACHE.get(`${MEM_PREFIX}last_dream`);
   495→  if (lastDreamRaw) stats.lastDream = JSON.parse(lastDreamRaw);
   496→
   497→  return stats;
   498→}

   500→// ============================================================
   501→// FONCTIONS INTERNES — Construction du matériel de rêve
   502→// ============================================================
   503→
   504→function buildDreamMaterial(dailyMemories, narratives) {
   505→  let material = `## MATÉRIEL DE RÊVE — ${dailyMemories.length} jours de mémoires\n\n`;
   506→
   507→  // Résumé chronologique compact
   508→  material += '### Chronologie (du plus ancien au plus récent)\n';
   509→  for (const mem of dailyMemories) {
   510→    const themesStr = mem.themes.slice(0, 3).join(' | ');
   511→    material += `- ${mem.date}: ${themesStr} (${mem.articleCount} articles, ${mem.sourceCount} sources)\n`;
   512→  }
   513→
   514→  // Thèmes avec fréquence
   515→  const themeFreq = {};
   516→  for (const mem of dailyMemories) {
   517→    for (const t of mem.themes) {
   518→      const norm = normalizeTheme(t);
   519→      themeFreq[norm] = (themeFreq[norm] || 0) + 1;
   520→    }
   521→  }
   522→  const frequent = Object.entries(themeFreq)
   523→    .sort((a, b) => b[1] - a[1])
   524→    .slice(0, 12);
   525→
   526→  material += '\n### Thèmes par fréquence\n';
   527→  for (const [key, count] of frequent) {
   528→    const bar = '█'.repeat(Math.min(count, 10));
   529→    material += `- ${key.replace(/_/g, ' ')} : ${bar} (${count}/${dailyMemories.length} jours)\n`;
   530→  }
   531→
   532→  // Sources les plus présentes
   533→  const sourceFreq = {};
   534→  for (const mem of dailyMemories) {
   535→    for (const s of mem.sources) {
   536→      sourceFreq[s] = (sourceFreq[s] || 0) + 1;
   537→    }
   538→  }
   539→  const topSources = Object.entries(sourceFreq)
   540→    .sort((a, b) => b[1] - a[1])
   541→    .slice(0, 8);
   542→
   543→  material += '\n### Sources les plus citées\n';
   544→  for (const [source, count] of topSources) {
   545→    material += `- ${source} : présente ${count}/${dailyMemories.length} jours\n`;
   546→  }
   547→
   548→  // Faits clés par jour
   549→  material += '\n### Faits marquants par jour\n';
   550→  for (const mem of dailyMemories.slice(-5)) {
   551→    material += `\n**${mem.date}**\n`;
   552→    for (const fact of mem.keyFacts.slice(0, 3)) {
   553→      material += `  - ${fact}\n`;
   554→    }
   555→  }
   556→
   557→  // Narratives actives
   558→  if (narratives?.active?.length > 0) {
   559→    material += '\n### Narratives en cours\n';
   560→    for (const nar of narratives.active.slice(0, 5)) {
   561→      material += `- **${nar.title}** : ${nar.dayCount} jours, dernier le ${nar.lastSeen}\n`;
   562→      if (nar.facts?.length > 0) {
   563→        material += `  Évolution: ${nar.facts[nar.facts.length - 1].fact}\n`;
   564→      }
   565→    }
   566→  }
   567→
   568→  // Acteurs récurrents
   569→  const actorFreq = {};
   570→  for (const mem of dailyMemories) {
   571→    if (mem.actors) {
   572→      for (const a of mem.actors) {
   573→        actorFreq[a] = (actorFreq[a] || 0) + 1;
   574→      }
   575→    }
   576→  }
   577→  const topActors = Object.entries(actorFreq)
   578→    .sort((a, b) => b[1] - a[1])
   579→    .slice(0, 8);
   580→
   581→  if (topActors.length > 0) {
   582→    material += '\n### Acteurs récurrents\n';
   583→    for (const [actor, count] of topActors) {
   584→      material += `- ${actor} : ${count} mentions\n`;
   585→    }
   586→  }
   587→
   588→  return material;
   589→}

   660→function ruleBasedDistillation(dailyMemories, narratives) {
   661→  // Patterns récurrents
   662→  const themeFreq = {};
   663→  for (const mem of dailyMemories) {
   664→    for (const t of mem.themes) {
   665→      const norm = normalizeTheme(t);
   666→      themeFreq[norm] = (themeFreq[norm] || 0) + 1;
   667→    }
   668→  }
   669→
   670→  const patterns = Object.entries(themeFreq)
   671→    .sort((a, b) => b[1] - a[1])
   672→    .slice(0, 6)
   673→    .map(([key, count]) => ({
   674→      type: 'recurring',
   675→      theme: key.replace(/_/g, ' '),
   676→      frequency: `${count}/${dailyMemories.length} jours`,
   677→      trend: count > dailyMemories.length * 0.5 ? 'stable' : 'emerging',
   678→      note: `Présent ${count} jours sur ${dailyMemories.length}`,
   679→    }));
   680→
   681→  // Narratives depuis la mémoire narratives
   682→  const narList = (narratives?.active || []).slice(0, 4).map(n => ({
   683→    title: n.title,
   684→    arc: `${n.dayCount} jours d'évolution`,
   685→    key_moments: n.facts?.slice(-2).map(f => f.fact) || [],
   686→    status: n.dayCount >= 5 ? 'peaked' : 'active',
   687→  }));
   688→
   689→  // Source dynamics
   690→  const sourceFreq = {};
   691→  for (const mem of dailyMemories) {
   692→    for (const s of mem.sources) {
   693→      sourceFreq[s] = (sourceFreq[s] || 0) + 1;
   694→    }
   695→  }
   696→  const sourceDynamics = Object.entries(sourceFreq)
   697→    .sort((a, b) => b[1] - a[1])
   698→    .slice(0, 5)
   699→    .map(([source, count]) => ({
   700→      source,
   701→      role: count > dailyMemories.length * 0.7 ? 'leader' : 'contributor',
   702→      notes: `Présent ${count} jours`,
   703→    }));
   704→
   705→  return {
   706→    patterns,
   707→    narratives: narList,
   708→    insights: [],
   709→    source_dynamics: sourceDynamics,
   710→    blind_spots: [],
   711→    macro_trends: [],
   712→    aiGenerated: false,
   713→    generatedAt: new Date().toISOString(),
   714→  };
   715→}

   817→function extractThemesFromReview(reviewText) {
   818→  const themes = [];
   819→  const nonThemes = new Set([
   820→    "L'ESSENTIEL DU JOUR", "ANALYSE THÉMATIQUE", "TENDANCES",
   821→    "CHIFFRES CLÉS", "À SURVEILLER", "ESSENTIEL", "TENDANCES & PERSPECTIVES",
   822→    "SOMMAIRE", "POINTS DE TENSION", "SOURCES", "POUR ALLER PLUS LOIN",
   823→    "APPROFONDIR", "NOTE FINALE",
   824→  ]);
   825→
   826→  // Format éditorial : **1. Titre éditorial** ou **1. Titre : sous-titre**
   827→  const editorialRegex = /^\*\*(\d+)\.[\s]+(.+?)(?:\s*[:：]\s*(.+?))?\*\*$/gm;
   828→  let match;
   829→  while ((match = editorialRegex.exec(reviewText)) !== null) {
   830→    const title = match[2].trim();
   831→    if (title.length > 5 && !nonThemes.has(title.toUpperCase())) {
   832→      themes.push(title);
   833→    }
   834→  }
   835→
   836→  // Format Smart Brevity legacy : **THÈME 1** suivi de :
   837→  const sectionRegex = /\*\*([^*\d]{3,40})\*\*\s*[:：]/g;
   838→  while ((match = sectionRegex.exec(reviewText)) !== null) {
   839→    const theme = match[1].trim();
   840→    if (!nonThemes.has(theme.toUpperCase()) && !themes.includes(theme)) {
   841→      themes.push(theme);
   842→    }
   843→  }
   844→
   845→  // Format axiom legacy : **Bold Axiom :**
   846→  const axiomRegex = /\*\*([^*]{2,25})\s*[:：]\*\*/g;
   847→  while ((match = axiomRegex.exec(reviewText)) !== null) {
   848→    const axiom = match[1].trim();
   849→    if (axiom.length > 3 && axiom.length < 25 && !nonThemes.has(axiom.toUpperCase()) && !themes.includes(axiom)) {
   850→      themes.push(axiom);
   851→    }
   852→  }
   853→
   854→  return [...new Set(themes)].slice(0, 12);
   855→}

   857→function extractKeyFacts(reviewText) {
   858→  const facts = [];
   859→  const seen = new Set();
   860→
   861→  // 1. Lignes à puces substantielles (30-250 chars)
   862→  const bulletRegex = /^-\s+(.{30,250})$/gm;
   863→  let match;
   864→  while ((match = bulletRegex.exec(reviewText)) !== null) {
   865→    const fact = match[1].trim();
   866→    // Exclure les lignes qui sont juste des titres de sections ou des liens
   867→    if (!fact.startsWith('**') && !fact.startsWith('http') && !seen.has(fact.substring(0, 50))) {
   868→      facts.push(fact);
   869→      seen.add(fact.substring(0, 50));
   870→    }
   871→  }
   872→
   873→  // 2. Phrases avec des chiffres précis dans les paragraphes (signal de faits durs)
   874→  const numberRegex = /[^.!?]*(?:\d{2,}|\d+\s*(?:millions?|milliards?|%|morts?|blessés?|personnes|jours|ans))[^.!?]*[.!?]/gi;
   875→  while ((match = numberRegex.exec(reviewText)) !== null) {
   876→    const fact = match[0].trim();
   877→    if (fact.length > 30 && fact.length < 250 && !seen.has(fact.substring(0, 50))) {
   878→      facts.push(fact);
   879→      seen.add(fact.substring(0, 50));
   880→    }
   881→  }
   882→
   883→  return [...new Set(facts)].slice(0, 15);
   884→}

   886→function extractActors(reviewText) {
   887→  const actors = new Set();
   888→  let match;
   889→
   890→  // 1. Noms propres en gras (2+ mots commençant par majuscule)
   891→  const boldNameRegex = /\*\*([A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ][a-zàâéèêëîïôùûüçæœ]+(?:\s+[A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ][a-zàâéèêëîïôùûüçæœ]+){0,3})\*\*/g;
   892→  while ((match = boldNameRegex.exec(reviewText)) !== null) {
   893→    const name = match[1].trim();
   894→    // Exclure les titres de sections (commençant par un chiffre ou trop longs)
   895→    if (!/^\d/.test(name) && name.length < 40 && name.split(/\s+/).length >= 2) {
   896→      actors.add(name);
   897→    }
   898→  }
   899→
   900→  // 2. Après « Selon », « D'après »
   901→  const selonRegex = /(?:Selon|D'après)\s+([^,.]{2,40})/gi;
   902→  while ((match = selonRegex.exec(reviewText)) !== null) {
   903→    const name = match[1].trim();
   904→    // Garder si ça ressemble à un nom propre (majuscule au début)
   905→    if (/^[A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ]/.test(name) && name.length > 2) {
   906→      actors.add(name);
   907→    }
   908→  }
   909→
   910→  // 3. Noms propres entre guillemets suivis d'une source
   911→  const quoteActorRegex = /«\s*[^»]{10,200}\s*»\s*[—–-]\s*([A-ZÀÂÉÈÊËÎÏÔÙÛÜÇÆŒ][a-zàâéèêëîïôùûüçæœ\s]{2,35})/g;
   912→  while ((match = quoteActorRegex.exec(reviewText)) !== null) {
   913→    actors.add(match[1].trim());
   914→  }
   915→
   916→  // 4. Liste élargie de figures connues (pattern flexible)
   917→  const knownFigures = [
   918→    'Macron', 'Barnier', 'Le Pen', 'Mélenchon', 'Scholz', 'Trump', 'Biden',
   919→    'Xi', 'Poutine', 'Starmer', 'Lula', 'Milei', 'Zelensky', 'Netanyahu',
   920→    'Mitsotakis', 'Maréchal', 'Darmanin', 'Attal', 'Borne', 'Castex',
   921→    'Von der Leyen', 'Blinken', 'Lavrov', 'Khamenei', 'Modi', 'Meloni',
   922→  ];
   923→  for (const figure of knownFigures) {
   924→    if (reviewText.includes(figure)) actors.add(figure);
   925→  }
   926→
   927→  return [...actors].slice(0, 15);
   928→}

   997→async function updateDayIndex(env, dateStr) {

  1001→  if (!index.days.includes(dateStr)) {
  1002→    index.days.push(dateStr);
  1003→    index.days.sort();
  1004→    // Garder max 60 jours
  1005→    index.days = index.days.slice(-60);

  1007→  }
  1008→  return index;

  1019→function computeThemeSimilarity(a, b) {
  1020→  const wordsA = new Set(a.split(/[\s_]+/));
  1021→  const wordsB = new Set(b.split(/[\s_]+/));
  1022→  if (wordsA.size === 0 || wordsB.size === 0) return 0;
  1023→  let intersection = 0;
  1024→  for (const w of wordsA) {
  1025→    if (wordsB.has(w)) intersection++;
  1026→  }
  1027→  return intersection / Math.max(wordsA.size, wordsB.size);
  1028→}
